[docs]    def load_data(self, environment_path, environment_log_path, reward_log_path, action_log_path):
        """Loads data from disk logs"""
        self.environment = Environment.load(environment_path)
        with open(environment_log_path) as file:
            self.environment_log = json.loads(file.read())
        with open(reward_log_path) as file:
            self.reward_log = json.loads(file.read())
        with open(action_log_path) as file:
            self.action_log = json.loads(file.read())
        self.radar_sweep_period = self.environment_log["radar_sweep_period"]
        self.total_steps = self.environment_log["total_steps"]
        self.environment = Environment.load(environment_path)



[docs]    def replay(self, burnin=0, mode="history"):
        """Replays the simulation visualisation, skipping the first burnin steps.
        When mode is set to 'history', a 2D viosualisation is produced. When mode is
        set to 'history_complete' a richer 3D visualisation is produced."""
        for s in range(self.total_steps):
            # Move time indexes by one step
            self.step()

            if s >= burnin:
                # Stepper visualises the aircraft trajectories
                super().render(mode)
