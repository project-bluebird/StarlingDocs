[docs]    def __init__(
        self,
        environment_path,
        environment_log_path,
        reward_log_path,
        action_log_path,
        aspect_ratio=1,
        scale=150,
    ):
        """
        Construct a Replayer instance.

        Parameters
        ----------
        environment_path: String
            Path where initial environment file (which initialised the simulation) was saved
        environment_log_path: String
            Path of environment log from simulation
        reward_log_path: String
            Path of rewards log from simulation
        action_log_path: String
            Path of actions log from simulation
        aspect_ratio: float
            Aspect ratio for all images produced - default = 1
        scale: float
            Scale factor for all images produced - default = 150
        """

        if environment_log_path is not None and reward_log_path is not None and action_log_path is not None:
            self.load_data(environment_path, environment_log_path, reward_log_path, action_log_path)
        else:
            raise ValueError("You need to provide environment_log_path, " "reward_log_path, action_log_path.")

        self.orig_airspace = self.environment.airspace

        self.callsign_list = []
        for callsign, aircraft in self.environment.aircraft.items():
            self.callsign_list.append(callsign)

        self.max_step_received = 0
        self.min_step_received = 1e8
        for callsign in self.callsign_list:
            self.max_step_received = max(self.max_step_received, max(self.environment_log[callsign]["step"]))
            self.min_step_received = min(self.min_step_received, min(self.environment_log[callsign]["step"]))

        # Rendering options
        self.aspect_ratio = aspect_ratio
        self.scale = scale
        self.centre = self.orig_airspace.fixes.centre()
        self.radar = Radar(self.centre, self.scale, self.aspect_ratio)



[docs]    def load_data(self, environment_path, environment_log_path, reward_log_path, action_log_path):
        """Loads data from disk logs"""
        self.environment = Environment.load(environment_path)
        with open(environment_log_path) as file:
            self.environment_log = json.loads(file.read())
        with open(reward_log_path) as file:
            self.reward_log = json.loads(file.read())
        with open(action_log_path) as file:
            self.action_log = json.loads(file.read())
        self.radar_sweep_period = self.environment_log["radar_sweep_period"]
        self.total_steps = self.environment_log["total_steps"]



[docs]    def regenerate_environment(self, step):
        """
        Takes in a step number from the program
        Has access to the original airspace and list of all callsigns
        in the environment from the class variable
        Uses the step to identify the datetime and the aircraft information
        Returns a new environment that can then be rendered

        Parameters
        ----------
        step: int
            Required parameter, step number of simulation
        """
        if step > self.max_step_received:
            raise ValueError(
                f"The highest step in the environment log file is {self.max_step_received}, the requested step was {step}"
            )

        time = None
        aircraft = {}
        for callsign in self.callsign_list:
            if step in self.environment_log[callsign]["step"]:
                indx = self.environment_log[callsign]["step"].index(step)
                time = self.environment_log[callsign]["datetime"][indx]
                aircraft[callsign] = self.make_aircraft(callsign, indx)
            if time is None:
                # Infer time from radar_sweep_period
                time = self.radar_sweep_period * step
                warnings.warn("Warning: Time is inferred from requested step and radar_sweep_period")
            environment = Environment(time, self.orig_airspace, aircraft)

        return environment



[docs]    def get_actions(self, step):
        """
        Extracts actions at given timestep, returns a list of strings
        """
        if step > self.max_step_received:
            raise ValueError(
                f"The highest step in the environment log file is {self.max_step_received}, the requested step was {step}"
            )

        actions = []
        for callsign in self.callsign_list:
            if callsign in self.action_log:
                if step in self.action_log[callsign]["step"]:
                    indx = self.action_log[callsign]["step"].index(step)
                    action = Action(
                        callsign, self.action_log[callsign]["kind"][indx], self.action_log[callsign]["value"][indx]
                    )
                    actions.append(str(action))
        return actions



[docs]    def make_aircraft(self, callsign, indx, fp=None):
        """
        Recieves the index of the time step
        Extracts the flight plan, lat, lon, fl, heading,
        speed, vspeed from the env_log_list (a class variable)
        Returns an Aircraft object

        Parameters
        ----------
        callsign: str
            Required parameter, Callsign (or name) of particular aircraft
        indx: int
            Required parameter, index correspinding to current step
        """
        fp_str = self.environment_log[callsign]["flight_plan"][indx]
        flight_plan = FlightPlan.from_str(fp_str)
        latitude = self.environment_log[callsign]["latitude"][indx]
        longitude = self.environment_log[callsign]["longitude"][indx]
        flight_level = self.environment_log[callsign]["flight_level"][indx]
        heading = self.environment_log[callsign]["heading"][indx]
        speed = self.environment_log[callsign]["calculated_ground_speed"][indx]
        vertical_speed = self.environment_log[callsign]["vertical_speed"][indx]
        aircraft = Aircraft(flight_plan, latitude, longitude, flight_level, heading, speed, vertical_speed)

        return aircraft



[docs]    def save_svg(
        self,
        step,
        basename="replayer",
        leading_zeros=0,
        blips=6,
        blip_color="gray",
    ):
        """
        Takes the environment object and outputs an svg

        Parameters
        ----------
        basename: str
            Optional parameter, filename for output svg. Default = replayer
        leading_zeros: int
            Optional parameter, number of leading zeros. Default = 0
        blips: int
            Optional parameter, number of circles (blips) following the aircraft
            emulating real-world radar features. Default = 6
        blip_color: str
            Optional parameter, named colour for radar blips. Default = gray
        """
        filename = f"tmp/{basename}_step_{step:0{leading_zeros}}"
        env = self.regenerate_environment(step)
        actions = self.get_actions(step)
        self.radar.draw_environment(env, radar_sweep_period=self.radar_sweep_period, blips=blips, blip_color=blip_color)
        self.radar.draw_action_queue(actions, env)
        self.radar.save(f"{filename}.svg")



[docs]    def save_png(
        self,
        step,
        basename="replayer",
        leading_zeros=0,
        blips=6,
        blip_color="gray",
    ):
        """
        If SVG is present, converts the svg to a png
        If no SVG present, then one is produced, then converted to png
        """
        filename = f"tmp/{basename}_step_{step:0{leading_zeros}}"
        chk_file = Path(f"{filename}.svg")
        if chk_file.is_file():
            cairosvg.svg2png(
                url=f"{filename}.svg", write_to=f"{filename}.png", output_width=900, background_color="white"
            )
        else:
            self.save_svg(
                step,
                basename,
                leading_zeros,
                blips,
                blip_color,
            )

            cairosvg.svg2png(
                url=f"{filename}.svg", write_to=f"{filename}.png", output_width=900, background_color="white"
            )



[docs]    def make_gif(
        self,
        burn_in=None,
        end_frame=None,
        basename="replayer",
        leading_zeros=0,
        blips=6,
        blip_color="gray",
        gif_output_name=None,
    ):
        """
        Stitches together sequential pngs into a gif
        if no initial and final frames are given, the entire history is produced
        default values for visualisation options

        Parameters
        ----------
        burn_in: int
            Optional parameter, step that will be the first frame of the gif.
            Default = first frame with aircraft in it (self.min_step_received)
        end_frame: int
            Optional parameter, step that will be the final frame of the gif
            Default = last frame with aircraft in it (self.max_step_received)
        gif_output_name: str
            Optional parameter, name of the output gif.
            Default = tmp/basename_step_first-step_to_end-step
        """
        if burn_in is None:
            burn_in = self.min_step_received
        if end_frame is None:
            end_frame = self.max_step_received

        for_gif_images = []
        for current_frame in range(burn_in, end_frame + 1):
            filename = f"tmp/{basename}_step_{current_frame:0{leading_zeros}}"
            chk_file = Path(f"{filename}.png")
            if not chk_file.is_file():
                self.save_png(
                    step=current_frame,
                    basename=basename,
                    leading_zeros=leading_zeros,
                    blips=blips,
                    blip_color=blip_color,
                )
            for_gif_images.append(imageio.imread(f"{filename}.png"))
        if gif_output_name is None:
            gif_output_name = f"tmp/{basename}_step_{burn_in:0{leading_zeros}}_to_{end_frame:0{leading_zeros}}"
        else:
            gif_output_name = gif_output_name

        imageio.mimsave(f"{gif_output_name}.gif", for_gif_images, loop=0)
