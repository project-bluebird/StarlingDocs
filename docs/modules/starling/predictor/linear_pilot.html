import typing

from ..core import Action, Environment, Trajectory, Pos2D

[docs]class LinearPilot(Predictor):
    """
    Evolves the Aircraft with constant speed and vertical speed along it's heading.
    If aircraft is in Route following mode, the heading is updated accordingly.

    Aircraft can be instructed to go directly to any Fix on its Route by giving a set_route command.
    Aircraft also responds to change_heading_* actions. If Aircraft is "on heading", it continues on
    that heading until an on_route or set_route action is issued, returning the Aircraft to Route
    following behaviour.
    """


[docs]    def __init__(self, environment: Environment, step_time: float, fix_proximity_threshold: float):
        """
        Construct a new instance.

        The internal step_time determines how

        If the aircraft position is <= fix_proximity_threshold,
            update what the next target Fix on Route is.
        """

        if step_time <= 0.0:
            raise ValueError("Step-time must be positive.")

        if fix_proximity_threshold <= 0.0:
            raise ValueError("Fix proximity threshold must be positive.")

        self.next_fix = {}
        for aircraft in environment.aircraft:
            self.next_fix[aircraft] = 1

        self.step_time = step_time
        self.fix_proximity_threshold = fix_proximity_threshold



[docs]    def get_target_pos(self, callsign: str, environment: Environment) -> Pos2D:
        """
        Get Pos2D of next Fix on Route.
        """

        aircraft = environment.aircraft[callsign]
        pos = aircraft.pos2d()
        target_pos = environment.airspace.fixes.places[aircraft.flight_plan.route.names[self.next_fix[callsign]]]

        return target_pos



[docs]    def future_trajectory(
        self, callsign: str, environment: Environment, dt: float, actions: typing.List[Action]
    ) -> Trajectory:
        """
        Predict the Trajectory evolution of a single Aircraft within the Environment for dt (delta t).
        Change heading or flight level if requested in Actions.
        """

        if dt <= 0.0:
            raise ValueError("Total time delta must be positive.")

        # keep track of updated
        self.environment = environment

        # Check for flight level and heading commands
        if (len(actions) > 0) and (self.next_fix[callsign] is not None):
            for action in actions:
                if action.callsign == callsign:
                    if action.kind == "change_flight_level_to":
                        self.environment.aircraft[callsign].requested_fl = action.value
                    else:
                        # heading action
                        bearing = self.get_bearing(callsign, action)
                        self.environment.aircraft[callsign].heading = bearing

        # Evolve Aircraft
        time = 0.0
        control_points = []
        while time < dt:

            # If last Fix on Route has been passed - stop
            if self.next_fix[callsign] is None:
                break

            step = min(self.step_time, dt - time)

            if self.environment.aircraft[callsign].flight_plan.entry.time <= (self.environment.time + time):

                self.update_bearing(callsign, step)
                self.update_flight_level(callsign, step)

            control_points.append(self.environment.aircraft[callsign].pos3d().pos4d(time + step))

            time += step

        if len(control_points) <= 2:
            return None

        return Trajectory(control_points)



[docs]    def get_bearing(self, callsign: str, action: Action) -> float:
        """
        Get bearing corresponding to issued action/command.
        """

        if action.kind == "change_heading_by":
            bearing = self.environment.aircraft[callsign].heading + action.value
            self.environment.on_route[callsign] = False
            self.environment.aircraft[callsign].requested_heading = bearing
        elif action.kind == "change_heading_to":
            bearing = action.value
            self.environment.on_route[callsign] = False
            self.environment.aircraft[callsign].requested_heading = bearing
        elif action.kind == "change_flight_level_by":
            flight_level = self.environment.aircraft[callsign].fl + action.value
            self.environment.aircraft[callsign].requested_fl = flight_level
        elif action.kind == "change_flight_level_to":
            flight_level = action.value
            self.environment.aircraft[callsign].requested_fl = flight_level
        elif action.kind == "on_route":
            # target pos is the next Fix on Route
            pos = self.environment.aircraft[callsign].pos2d()
            target_pos = self.get_target_pos(callsign, self.environment)
            bearing = pos.bearing_to(target_pos)
            self.environment.on_route[callsign] = True
        elif action.kind == "set_route":
            pos = self.environment.aircraft[callsign].pos2d()
            target_fix = action.value
            target_pos = self.environment.airspace.fixes.places[target_fix]
            bearing = pos.bearing_to(target_pos)
            self.environment.on_route[callsign] = True
            # save what the next Fix is
            route_pos = self.environment.aircraft[callsign].flight_plan.route.names.index(target_fix)
            self.next_fix[callsign] = route_pos
        else:
            raise ValueError(f"Unrecognised Action kind: {action.kind}")

        return bearing



[docs]    def update_bearing(self, callsign, step):
        """
        Update aircraft bearing.

        If Aircraft is following its Route:
           - make sure it has the right heading (next Fix)
        Else Aircraft should just continue on its current heading:
           - no update needed

        Update what the next target Fix on Route is if necessary.
        """

        pos = self.environment.aircraft[callsign].pos2d()

        # if route following - get heading
        if self.environment.on_route[callsign]:
            target_pos = self.get_target_pos(callsign, self.environment)
            autopilot_heading = pos.bearing_to(target_pos)
            self.environment.aircraft[callsign].heading = autopilot_heading

        dx = self.environment.aircraft[callsign].speed * (step / 3600.0)
        new_pos = pos.forward(dx, self.environment.aircraft[callsign].heading)

        self.environment.aircraft[callsign].lat = new_pos.lat
        self.environment.aircraft[callsign].lon = new_pos.lon

        # if route_folliwng - update next_fix
        if self.environment.on_route[callsign]:
            target_pos = self.get_target_pos(callsign, self.environment)
            if new_pos.distance(target_pos) <= self.fix_proximity_threshold:
                self.next_fix[callsign] += 1
                if self.next_fix[callsign] >= len(self.environment.aircraft[callsign].flight_plan.route.names):
                    self.next_fix[callsign] = None


    def update_flight_level(self, callsign, step):

        # assume climd/descend rate of 20 FL a minute
        rate = 20 / (60 / step)

        fl = self.environment.aircraft[callsign].fl
        req_fl = self.environment.aircraft[callsign].requested_fl

        if fl > req_fl:
            # descend
            self.environment.aircraft[callsign].fl -= rate
        elif fl < req_fl:
            # climb
            self.environment.aircraft[callsign].fl += rate
