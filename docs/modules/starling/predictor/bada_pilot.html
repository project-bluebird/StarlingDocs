[docs]class BadaPilot(Predictor):
    """
    Evolves Aircraft with constant speed and vertical speed along it's heading. Responds to lateral and vertical Actions.
    If aircraft is in Route following mode, the heading is updated when a Fix is passed. If Aircraft is given a `change_heading_*`
    instruction, it continues on that heading until an `on_route` or `set_route` action is issued.

    The rate of climb or descent is determined using a BADA model implementation
    """

        """
        Construct a new instance.

        Parameters
        ----------
        step_time: float
            The internal step time of the predictor. The time taken between each control point within the returned Trajectory is step_time.
        fix_proximity_threshold: float
            When an Aircraft is <= fix_proximity_threshold distance from the next Fix on its Route, the Fix is considered to have been passed.
            The next target Fix on Route for the Aircraft is then updated.
        """

        super(BadaPilot, self).__init__(step_time, fix_proximity_threshold)



[docs]    def get_target_pos(self, callsign: str) -> Pos2D:
        """
        Get Pos2D of next Fix on Route.

        Parameters
        ----------
        callsign: str
            An Aircraft callsign

        Returns
        -------
        Pos2D
            Position of the next Fix on the Aircraft Route
        """

        aircraft = self.environment.aircraft[callsign]
        target_pos = self.environment.airspace.fixes.places[aircraft.flight_plan.route.names[self.next_fix[callsign]]]

        return target_pos



[docs]    def future_trajectory(
        self, callsign: str, environment: Environment, dt: float, actions: typing.List[Action]
    ) -> Trajectory:
        """
        Predict the Trajectory evolution of a single Aircraft within the Environment for dt amount of time.
        Change heading or flight level if requested in Actions.

        Parameters
        ----------
        callsign: str
            Aircraft callsign
        environment : Environment
            An Environment instance
        dt: float
            The amount of time [sec] for which to evolve the Aircraft Trajectory
        actions: list[Action]
            List of Actions issued to the Aircraft. Action kind can be one of:
                - `on_route`: automatically follow filed Route
                - `set_route`: go directly to named Fix on Route
                - `change_heading_to`: change heading to the specified degrees
                - `change_heading_by`: change heading by given degrees
                - `change_flight_level_to`: go to the specified flight level
                - `change_flight_level_by`: descend/ascend by specified flight levels

        Returns
        -------
        Trajectory
            The Aircraft Trajectory
        """

        if dt <= 0.0:
            raise ValueError("Total time delta must be positive.")

        # keep track of updates
        # note: use copy here to avoid changing internal state of environment
        self.environment = copy.deepcopy(environment)

        # Check for flight level and heading commands
        if (len(actions) > 0) and (self.next_fix[callsign] is not None):
            for action in actions:
                if action.callsign == callsign:
                    if action.kind == "change_flight_level_to" or action.kind == "change_flight_level_by":
                        self.process_vertical_actions(callsign, action)
                    else:
                        self.process_lateral_actions(callsign, action)

        # Evolve Aircraft
        time = 0.0
        control_points = []
        while time < dt:

            # If last Fix on Route has been passed - stop
            if self.next_fix[callsign] is None:
                break

            step = min(self.step_time, dt - time)

            self.update_heading(callsign, step)
            self.update_flight_level(callsign, step)

            time += step
            control_points.append(self.environment.aircraft[callsign].pos3d().pos4d(environment.time + time))

        if len(control_points) < 2:
            return None

        return Trajectory(control_points)



[docs]    def process_lateral_actions(self, callsign: str, action: Action):
        """
        Update `heading` and `requested_heading` according to issued action/command.

        Parameters
        ----------
        callsign: str
            Aircraft callsign
        action: Action
            Action issued to the Aircraft
        """

        if action.kind == "change_heading_by":
            heading = self.environment.aircraft[callsign].heading + action.value
            self.environment.on_route[callsign] = False
            self.environment.aircraft[callsign].requested_heading = heading % 360

        elif action.kind == "change_heading_to":
            heading = action.value
            self.environment.on_route[callsign] = False
            self.environment.aircraft[callsign].requested_heading = heading % 360

        elif action.kind == "on_route":
            # target pos is the next Fix on Route
            pos = self.environment.aircraft[callsign].pos2d()
            target_pos = self.get_target_pos(callsign)
            heading = pos.bearing_to(target_pos)
            self.environment.on_route[callsign] = True
            self.environment.aircraft[callsign].requested_heading = None

        elif action.kind == "set_route":
            pos = self.environment.aircraft[callsign].pos2d()
            target_fix = action.value
            target_pos = self.environment.airspace.fixes.places[target_fix]
            heading = pos.bearing_to(target_pos)
            self.environment.on_route[callsign] = True
            self.environment.aircraft[callsign].requested_heading = None
            # save what the next Fix is
            fix_idx = self.environment.aircraft[callsign].flight_plan.route.names.index(target_fix)
            self.next_fix[callsign] = fix_idx

        else:
            raise ValueError(f"Unrecognised Action kind: {action.kind}")

        self.environment.aircraft[callsign].heading = heading % 360



[docs]    def process_vertical_actions(self, callsign: str, action: Action):
        """
        Update `requested_fl` according to issued action/command.

        Parameters
        ----------
        callsign: str
            Aircraft callsign
        action: Action
            Action issued to the Aircraft
        """

        if action.kind == "change_flight_level_by":
            flight_level = self.environment.aircraft[callsign].fl + action.value

        elif action.kind == "change_flight_level_to":
            flight_level = action.value

        else:
            raise ValueError(f"Unrecognised Action kind: {action.kind}")

        self.environment.aircraft[callsign].requested_fl = flight_level



[docs]    def update_heading(self, callsign, step):
        """
        Move aircraft laterally and update it's heading as required.
        If Aircraft is following its Route:
           - make sure it has the right heading (next Fix)
           - update what the next target Fix on Route is if target Fix is passed
        Else Aircraft should just continue on its current heading:
           - no update needed

        Parameters
        ----------
        callsign: str
            Aircraft callsign
        step: float
            Amount of time [sec] to evolve the Aircraft for
        """

        pos = self.environment.aircraft[callsign].pos2d()

        # if route following - get heading
        if self.environment.on_route[callsign]:
            target_pos = self.get_target_pos(callsign)
            autopilot_heading = pos.bearing_to(target_pos)
            self.environment.aircraft[callsign].heading = autopilot_heading

        dx = self.environment.aircraft[callsign].speed * (step / 3600.0)
        new_pos = pos.forward(dx, self.environment.aircraft[callsign].heading)

        self.environment.aircraft[callsign].lat = new_pos.lat
        self.environment.aircraft[callsign].lon = new_pos.lon

        # if route_following - update next_fix
        if self.environment.on_route[callsign]:
            target_pos = self.get_target_pos(callsign)

            if new_pos.distance(target_pos) <= self.fix_proximity_threshold:
                self.next_fix[callsign] += 1

                if self.next_fix[callsign] >= len(self.environment.aircraft[callsign].flight_plan.route.names):
                    self.next_fix[callsign] = None



[docs]    def update_flight_level(self, callsign: str, step: float):
        """
        Update aircraft's flight level based on BADA model. Here we assume
        an ISA (International Standard Atmosphere) at position of the aircraft.
        In future might consider different BADA atmospheres, i.e. delta_T and
        delta_P values other than zero.

        Parameters
        ----------
        callsign: str
            Aircraft callsign
        step: float
            Amount of time [sec] to evolve the Aircraft for
        """
        atmos = Atmosphere(delta_T=0.0, delta_P=0.0)
        fl = self.environment.aircraft[callsign].fl
        req_fl = self.environment.aircraft[callsign].requested_fl
        if fl > req_fl:
            # descend
            delta_Hp = bada.change_in_pressure_height_and_mass(
                aircraft=self.environment.aircraft[callsign], atmos=atmos, aircraft_state="descent", time_interval=step
            )[0]
            delta_fl = delta_Hp * convert.M_TO_FL
            self.environment.aircraft[callsign].fl += delta_fl
            if self.environment.aircraft[callsign].fl < req_fl:
                self.environment.aircraft[callsign].fl = req_fl
        elif fl < req_fl:
            # climb
            delta_Hp = bada.change_in_pressure_height_and_mass(
                aircraft=self.environment.aircraft[callsign], atmos=atmos, aircraft_state="climb", time_interval=step
            )[0]
            delta_fl = delta_Hp * convert.M_TO_FL
            self.environment.aircraft[callsign].fl += delta_fl
            if self.environment.aircraft[callsign].fl > req_fl:
                self.environment.aircraft[callsign].fl = req_fl

        if self.environment.aircraft[callsign].fl < 0:
            print("Flight level in bada_pilot update_fl is negative!")
            print("FL is {} and requested FL is {}".format(self.environment.aircraft[callsign].fl, req_fl))
