    """
    Does not respond to Actions,
    evolves the Aircraft with constant speed and infinite acceleration along it's Route.

[docs]    def __init__(self, environment: Environment, step_time: float, fix_proximity_threshold: float):
        """
        Construct a new instance.
        """

        if step_time <= 0.0:
            raise ValueError("Step-time must be positive.")

        if fix_proximity_threshold <= 0.0:
            raise ValueError("Fix proximity threshold must be positive.")

        self.next_fix = {}
        for aircraft in environment.aircraft:
            self.next_fix[aircraft] = 1

        self.step_time = step_time
        self.fix_proximity_threshold = fix_proximity_threshold



[docs]    def future_trajectory(
        self, callsign: str, environment: Environment, dt: float, actions: typing.List[Action]
    ) -> Trajectory:
        """
        Predict the Trajectory evolution of a single Aircraft within the Environment.
        """

        if dt <= 0.0:
            raise ValueError("Total time delta must be positive.")

        aircraft = environment.aircraft[callsign]

        time = 0.0
        control_points = []
        while time < dt:
            if self.next_fix[callsign] is None:
                break

            step = min(self.step_time, dt - time)

            if aircraft.flight_plan.entry.time <= (environment.time + time):

                pos = aircraft.pos2d()
                target_pos = environment.airspace.fixes.places[
                    aircraft.flight_plan.route.names[self.next_fix[callsign]]
                ]

                aircraft.heading = pos.bearing_to(target_pos)

                dx = aircraft.speed * (step / 3600.0)
                new_pos = pos.forward(dx, aircraft.heading)

                aircraft.lat = new_pos.lat
                aircraft.lon = new_pos.lon

                if new_pos.distance(target_pos) <= self.fix_proximity_threshold:
                    self.next_fix[callsign] += 1

                    if self.next_fix[callsign] >= len(aircraft.flight_plan.route.names):
                        self.next_fix[callsign] = None

            control_points.append(aircraft.pos3d().pos4d(time + step))

            time += step

        if len(control_points) <= 2:
            return None

        return Trajectory(control_points)
