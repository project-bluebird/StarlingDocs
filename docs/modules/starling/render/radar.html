[docs]    def __init__(self, centre: Pos2D, scale: float, aspect_ratio: float):
        """
        Initialise the Radar to observe a central location and a given width scale [nmi] with a given aspect-ratio (=width/height).
        """

        if aspect_ratio <= 0.0:
            raise ValueError("Aspect ratio must be positive.")

        self.centre = centre
        self.scale = scale
        self.aspect_ratio = aspect_ratio

        self.buffer = ""

        self.counter = 0

        self.step_counter = 0

        self.width = self.scale * convert.NMI_TO_DEG
        self.height = self.width / self.aspect_ratio



[docs]    def save(self, filename: str):
        """
        Save the drawing to an SVG image then clear the buffer.
        """

        width = self.scale * convert.NMI_TO_DEG
        height = width / self.aspect_ratio

        min_x = self.centre.lon - (width / 2.0)
        min_y = self.centre.lat - (height / 2.0)

        with open(filename, "w") as file:
            file.write(
                f"""<svg viewBox="{min_x} {min_y} {width} {height}" xmlns="http://www.w3.org/2000/svg" font-family="monospace" transform="scale(1, -1)">\n{self.buffer}</svg>"""
            )

        self.buffer = ""

[docs]    def draw_pos2d(self, location: Pos2D, label: str = ""):
        """
        Draw a location.
        """

        self.buffer += svg.circle(
            location.lon,
            location.lat,
            self.scale * 1e-4,
            fill="none",
            stroke="chocolate",
            stroke_width=(self.scale * 1e-4),
        )

        if len(label) > 0:
            self.buffer += svg.text(label, location.lon, location.lat - (self.scale * 4e-4), self.scale * 2e-4)



[docs]    def draw_pos3d(self, position: Pos3D, label: str = ""):
        """
        Draw a position.
        """

        f = max(min(position.fl / 400.0, 1.0), 0.0)

        red = 0.0
        green = 1.0 - f
        blue = f

        self.buffer += svg.circle(
            position.lon,
            position.lat,
            self.scale * 1e-4,
            fill=f"rgb({(red * 100)}%, {(green * 100)}%, {(blue * 100)}%)",
            stroke="black",
            stroke_width=(self.scale * 0.2e-4),
        )

        if len(label) > 0:
            self.buffer += svg.text(label, position.lon, position.lat - (self.scale * 4e-4), self.scale * 2e-4)



[docs]    def draw_pos4d(self, control_point: Pos4D, label: str = ""):
        """
        Draw a control point.
        """

        f = max(min(control_point.fl / 400.0, 1.0), 0.0)

        r = 0.0
        g = 1.0 - f
        b = f
        fill = f"rgb({r * 100}%, {g * 100}%, {b * 100}%)"

        t = math.fmod(control_point.time, 900.0) / 900.0
        stroke = f"hsl({int(t * 360)}, 70%, 40%)"

        self.buffer += svg.circle(
            control_point.lon,
            control_point.lat,
            self.scale * 0.25e-4,
            fill=fill,
        )
        self.buffer += svg.circle(
            control_point.lon,
            control_point.lat,
            self.scale * 0.25e-4,
            fill="none",
            stroke=stroke,
            stroke_width=(self.scale * 0.125e-4),
        )

        if len(label) > 0:
            self.buffer += svg.text(
                label, control_point.lon, control_point.lat - (self.scale * 4e-4), self.scale * 2e-4
            )



[docs]    def draw_environment_time(self, environment: Environment):
        """
        Add environment time and step count to visualisation
        """
        size = self.scale * 2e-4
        
        time_location = Pos2D(
            self.centre.lon - (3.0*self.height/8.0), 
            self.centre.lat + (3.0*self.width/8.0)
            )

        time_label = "Environment Time:" + tm.strftime('%H:%M:%S', tm.gmtime(environment.time))

        self.buffer += svg.text(time_label, time_location.lon, time_location.lat, size)

        step_location = Pos2D(
            self.centre.lon + (3.0*self.height/8.0),
            self.centre.lat + (3.0*self.width/8.0)
        )

        step_label = "Step:" + str(self.step_counter)

        self.buffer += svg.text(step_label, step_location.lon, step_location.lat, size)

        self.step_counter += 1

[docs]    def draw_route(self, route: Route, fixes: Fixes):
        """
        Draw a route.
        """

        for location_a, location_b in zip(route.names, route.names[1:]):
            start = fixes.places[location_a]
            end = fixes.places[location_b]
            self.buffer += svg.line(
                start.lon,
                start.lat,
                end.lon,
                end.lat,
                stroke_width=(self.scale * 1e-4),
                stroke="#DDDDDD",
                stroke_linecap="round",
            )

        end = fixes.places[route.names[-1]]
        self.buffer += svg.circle(end.lon, end.lat, self.scale * 0.75e-4, fill="gold")



[docs]    def draw_plan(self, plan: Plan):
        """
        Draw a plan.
        """

        for start, end in zip(plan.positions, plan.positions[1:]):
            self.counter += 1
            id = f"plan_grad_{self.counter}"

            f = max(min(start.fl / 400.0, 1.0), 0.0)
            start_col = f"rgb(0%, {f * 100}%, {(1 - f) * 100}%)"

            g = max(min(end.fl / 400.0, 1.0), 0.0)
            end_col = f"rgb(0%, {g * 100}%, {(1 - g) * 100}%)"

            self.buffer += svg.gradient(id, start.lon, start.lat, end.lon, end.lat, [end_col, start_col], [0, 1])
            self.buffer += svg.line(
                start.lon,
                start.lat,
                end.lon,
                end.lat,
                stroke_width=(self.scale * 1e-4),
                stroke=f'url("#{id}")',
                stroke_dasharray=f"{self.scale * 4e-4} {self.scale * 1.5e-4}",
                stroke_linecap="round",
            )

        self.buffer += svg.circle(plan.positions[0].lon, plan.positions[0].lat, self.scale * 0.75e-4, fill="black")
        self.buffer += svg.circle(plan.positions[-1].lon, plan.positions[-1].lat, self.scale * 0.75e-4, fill="gold")

[docs]    def draw_volume(self, volume: Volume):
        """
        Draw a volume.
        """

        points = []
        for coord in volume.area.boundary.exterior.coords:
            points.append((coord[0], coord[1]))

        f = max(min((volume.min_fl + volume.max_fl) / 800.0, 1.0), 0.0)
        fill = f"rgba(0%, {(1 - f) * 100}%, {f * 100}%, 0.1)"

        self.buffer += svg.polygon(points, fill=fill)

[docs]    def draw_aircraft(self, aircraft: Aircraft, labels: typing.List[str] = []):
        """
        Draw an aircraft.
        """

        wings = [
            [1.5, 0],
            [-1, 2.5],
            [-1.5, 2.5],
            [-0.5, 0],
            [-1.5, -2.5],
            [-1, -2.5],
        ]
        fuselage = [
            [-2, 0.5],
            [2, 0.5],
            [2.5, 0],
            [2, -0.5],
            [-2, -0.5],
        ]
        tail = [
            [-1, 0],
            [-2, 1],
            [-2.5, 1],
            [-2, 0],
            [-2.5, -1],
            [-2, -1],
        ]

        s = svg.polygon(wings, fill="red", stroke="black", stroke_width=0.1)
        s += svg.polygon(fuselage, fill="gainsboro", stroke="black", stroke_width=0.1)
        s += svg.polygon(tail, fill="red", stroke="black", stroke_width=0.1)

        self.buffer += svg.transform(s, self.scale * 1e-4, -(aircraft.heading - 90), f"{aircraft.lon},{aircraft.lat}")

        if len(labels) > 0:
            labels = svg.whitespace(labels)
            lat_diff = self.scale * 4e-4
            lon_diff = lat_diff
            size = self.scale * 2e-4
            for i, label in enumerate(labels):
                if(label.count("FL")) > 1:
                    self.buffer += svg.fl_gradient(label, aircraft.lon + lon_diff, aircraft.lat + (2*lat_diff) - i * size, size)
                else:
                    self.buffer += svg.text(label, aircraft.lon + lon_diff, aircraft.lat + (2*lat_diff) - i * size, size)



[docs]    def draw_environment(self, environment: Environment):
        """
        Draw the environment.
        """
        
        self.draw_environment_time(environment)
        
        for sector in environment.airspace.sectors.values():
            for volume in sector.volumes:
                self.draw_area(volume.area)

        for sector in environment.airspace.sectors.values():
            for volume in sector.volumes:
                self.draw_volume(volume)

        for name, aircraft in environment.aircraft.items():
            if (aircraft.flight_plan.entry.time <= environment.time) and (
                environment.time <= aircraft.flight_plan.exit.time
            ):
                self.draw_route(aircraft.flight_plan.route, environment.airspace.fixes)

        for name, aircraft in environment.aircraft.items():
            if (aircraft.flight_plan.entry.time <= environment.time) and (
                environment.time <= aircraft.flight_plan.exit.time
            ):
                end = environment.airspace.fixes.places[aircraft.flight_plan.route.names[-1]]
                self.buffer += svg.circle(end.lon, end.lat, self.scale * 0.75e-4, fill="gold")

        self.draw_fixes(environment.airspace.fixes)

        for name, aircraft in environment.aircraft.items():
            if (aircraft.flight_plan.entry.time <= environment.time) and (
                environment.time <= aircraft.flight_plan.exit.time
            ):
                labels = [name, f"FL{int(aircraft.fl)} FL{int(aircraft.requested_fl)}", f"C{int(aircraft.flight_plan.exit.fl)}"]
                self.draw_aircraft(aircraft, labels)




[docs]    def draw_actions(self, action: typing.List[str]) :
        """
        Adds current actions to the bottom left of image
        """
        size = self.scale * 2e-4

        action_location = Pos2D(
            self.centre.lon - (3.0*self.height/8.0), 
            self.centre.lat - (3.0*self.width/8.0)
            )

        if len(action) > 0:
            lat_diff = self.scale * 4e-4
            size = self.scale * 2e-4
            for _ in range(len(action)):
                label = str(action[_])
                self.buffer += svg.text(label, action_location.lon, action_location.lat - lat_diff - _ * size, size)
