[docs]    def __init__(self, lat: float, lon: float, fl: float, time: float):
        """
        Construct a new instance.
        """

        if (lat < -90.0) or (lat > 90.0):
            raise ValueError("Latitude must be in the range [-90, +90] degrees.")

        if (lon < -180.0) or (lon > 180.0):
            raise ValueError("Longitude must be in the range [-180, +180] degrees.")

        if fl < 0.0:
            raise ValueError("Flight level must be non-negative.")

        if time < 0.0:
            raise ValueError("Time must be non-negative.")

        self.location = np.array([lat, lon, fl, time])



[docs]    @staticmethod
    def from_str(s: str) -> Pos4D:
        """
        Construct a new instance from a string representation.
        """

        s.strip()
        [lat, lon, fl, time] = s.split()

        return Pos3D.from_str(f"{lat} {lon} {fl}").pos4d(convert.time_str(time))



[docs]    @staticmethod
    def from_array(a: np.ndarray) -> Pos4D:
        """
        Construct a new instance from a numpy array.
        """

        if a.shape != (4,):
            raise ValueError(f"Expected array shape (4,), given: {a.shape}")

        lat, lon, fl, time = a
        return Pos4D(lat, lon, fl, time)

    def __str__(self) -> str:
        """
        Create a human readable representation of the instance.
        """

        timestamp = time.gmtime(self.time)
        time_str = time.strftime("%H:%M:%S", timestamp)

        return f"{self.pos3d()} {time_str}"

    def __round__(self, precision: int) -> Pos4D:
        """
        Round the control-point elements to the given precision.
        """

        return Pos4D(
            round(self.lat, precision),
            round(self.lon, precision),
            round(self.fl, precision),
            round(self.time, precision),
        )

[docs]    def forward(self, dist: float, heading: float, speed: float) -> Pos4D:
        """
        Determine the four-dimensional control-point
        a given distance away [nmi] following a given heading [deg] at a given constant speed [kts].
        Flight level is maintained.
        """

        proj_pos2d = self.pos2d().forward(dist, heading)

        return Pos4D(proj_pos2d.lat, proj_pos2d.lon, self.fl, self.time + (dist / speed) * convert.HRS_TO_SEC)



[docs]    def bearing_to(self, other: Pos4D) -> float:
        """
        Calculate the bearing [deg] to another location.
        """

        _, back_azimuth, _ = Pos2D.geod.inv(other.lon, other.lat, self.lon, self.lat)

        return back_azimuth % 360.0



[docs]    def distance(self, other: Pos4D) -> float:
        """
        Calculate the geodesic distance [nmi] to another position.
        Flight level is considered.
        """

        return self.pos3d().distance(other.pos3d())



[docs]    def interval(self, other: Pos4D, speed: float) -> float:
        """
        Calculate the time interval [sec] to another position given a constant speed [kt].
        Flight level is considered.
        """

        return self.distance(other) / speed
