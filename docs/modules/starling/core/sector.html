[docs]    @staticmethod
    def from_json(s: str) -> Sector:
        """
        Construct a new instance from JSON representation.
        """

        volumes = []
        for volume in json.loads(s):
            volumes.append(Volume.from_json(json.dumps(volume)))

        return Sector(volumes)

[docs]    def contains(self, position: Pos3D) -> bool:
        """
        Determine if a given position is contained within the Sector.
