[docs]    def centre(self) -> Pos2D:
        """
        Get the geometric centre of the Fixes.
        """

        centre = Pos2D(0.0, 0.0)
        for location in self.places.values():
            centre.lat += location.lat
            centre.lon += location.lon

        centre.lat /= len(self.places)
        centre.lon /= len(self.places)

        return centre



[docs]    def delta_lat(self) -> float:
        """
        Get the latitudinal delta of the Fixes.
        """

        from ..utility import convert

        if len(self.places) < 2:
            return 0.0

        min = math.inf
        max = -math.inf
        for place in self.places.values():
            if place.lat < min:
                min = place.lat
            if place.lat > max:
                max = place.lat

        return (max - min) * convert.DEG_TO_NMI
