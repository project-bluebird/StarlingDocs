from __future__ import annotations
import time



[docs]class Contract:
    """
    Entry or exit agreement, i.e. a flight level and time [sec] when entry/exit should happen.
    """


[docs]    def __init__(self, fl: float, time: float):
        """
        Construct a new instance.

        Parameters
        ----------
        fl: float
            The contract flight level of entry/exit.
        time: float
            The contract time in seconds of entry/exit.

        Examples
        --------
        >>> entry_contract = Contract(190, 720)
        """

        if fl < 0.0:
            raise ValueError("Flight level must be non-negative.")

        if time < 0.0:
            raise ValueError("Time must be non-negative.")

        self.fl = fl
        self.time = time



[docs]    @staticmethod
    def from_str(s: str) -> Contract:
        """
        Construct a new instance from string representation.

        Parameters
        ----------
        s: str
            A string representation of Contract.

        Returns
        ----------
        Contract

        Examples
        ----------
        >>> contract = Contract("190.0FL 00:12:00")
        """

        s.strip()

        s = s.split()
        fl = s[0]
        time = s[1]

        if fl[-2:] != "FL":
            raise ValueError(f"Invalid flight level string: {fl}")

        [hr, min, sec] = time.split(":")
        time = (3600.0 * float(hr)) + (60.0 * float(min)) + float(sec)

        return Contract(float(fl[:-2]), time)


    def __str__(self) -> str:
        """
        Create a human readable representation of the instance.

        Returns
        ----------
        str
            A string representation of a contract (e.g., "190.0FL 00:12:00").
        """

        time_str = time.strftime("%H:%M:%S", time.gmtime(self.time))

        return f"{self.fl}FL {time_str}"
