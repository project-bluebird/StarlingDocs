from __future__ import annotations
import json

from .contract import Contract
from .route import Route



[docs]class FlightPlan:
    """
    A static record of a flight plan, which contains entry and exit Contracts and a Route.
    """


[docs]    def __init__(self, entry: Contract, exit: Contract, route: Route):
        """
        Construct a new instance.

        Parameters
        ----------
        entry: Contract
            A contract defining the flight level and time [sec] of entry.
        exit: Contract
            A contract defining the flight level and time [sec] of exit.
        route: Route
            The route to be followed.

        Examples
        --------
        >>> flight_plan = FlightPlan(Contract(160, 0.0), Contract(200, 900.0), Route(["ALFA", "BRAVO", "CHAR"]))
        """

        if entry.time < 0.0:
            raise ValueError("Entry time must be non-negative.")

        if entry.time > exit.time:
            raise ValueError("Exit time must be after entry time.")

        self.entry = entry
        self.exit = exit

        self.route = route



[docs]    @staticmethod
    def from_json(s: str) -> FlightPlan:
        """
        Construct a new instance from string in JSON format.

        Parameters
        ----------
        s: str
            A string representation of a FlightPlan in JSON/dictionary format.

        Returns
        ----------
        FlightPlan

        Examples
        ----------
        >>> FlightPlan.from_json(
        >>>     '''
        >>>     {
        >>>         "route": [
        >>>             "ALFA",
        >>>             "BRAVO",
        >>>             "CHAR"
        >>>         ],
        >>>         "entry": "160FL 00:00:00",
        >>>         "exit": "200FL 00:15:00"
        >>>     }
        >>>     '''
        >>> )
        """

        data = json.loads(s)

        entry = Contract.from_str(f"{data['entry']}")
        exit = Contract.from_str(f"{data['exit']}")
        route = Route.from_json(json.dumps(data["route"]))

        return FlightPlan(entry, exit, route)

[docs]    @staticmethod
    def from_str(s: str) -> FlightPlan:
        """
        Construct a new instance from a string representation. This is different to
        the from_json method above.

        Parameters
        ----------
        s: str
            A string representation of a flight plan.

        Returns
        ----------
        FlightPlan

        Examples
        ----------
        >>> FlightPlan.from_str(
        >>>     '''
        >>>     [190.0FL 00:12:00] > FIYRE > EARTH > AIR > SPIRT > [190.0FL 01:47:59]
        >>>     '''
        >>> )
        """
        entry = Contract.from_str(s[s.find("[") + 1 : s.find("]")])

        exit = Contract.from_str(s[s.rfind("[") + 1 : s.rfind("]")])

        route_str = s[s.find("]") + 1 : s.rfind("[")].replace(">", " ")
        route = Route(route_str.split())

        return FlightPlan(entry, exit, route)
