[docs]    def __init__(self, boundary: list[Pos2D]):
        """
        Construct a new instance.
        """

        if len(boundary) < 3:
            raise ValueError("Boundary must contain at least three locations.")

        self.boundary = Polygon([[location.lon, location.lat] for location in boundary])



[docs]    @staticmethod
    def from_json(s: str) -> Area:
        """
        Construct a new instance from JSON representation.
        """

        boundary = []
        for location in json.loads(s):
            boundary.append(Pos2D.from_str(location))

        return Area(boundary)

        """
        Get the data as a serialisable list.
        """

        return list(map(lambda p: f"{Pos2D(p[1], p[0])}", self.boundary.exterior.coords))[:-1]

[docs]    def contains(self, location: Pos2D, epsilon: float = epsilon) -> bool:
        """
        Determine if the given location is contained within the Area including the boundary (within epsilon tolerance).
        """

        return self.boundary.buffer(epsilon).intersects(Point(location.lon, location.lat))



[docs]    def on_boundary(self, location: Pos2D, epsilon: float = epsilon) -> bool:
        """
        Determine if the given location is on the Area boundary (within epsilon tolerance).
        """

        # NOTE: self.boundary is a Polygon, to get the boundary (LineString) you have to call self.boundary.boundary
        return self.boundary.boundary.distance(Point(location.lon, location.lat)) < epsilon
