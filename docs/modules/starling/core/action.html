    def from_str(s: str) -> Action:
        """
        Construct a new instance from string representation.
        """

        s.strip()

        s = s.split()
        callsign = s[0]
        kind = s[1]
        value = " ".join(s[2:])

        return Action(callsign, kind, json.loads(value))


    def __str__(self) -> str:
