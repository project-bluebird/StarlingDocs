[docs]    def __init__(self, callsign: str, kind: str, value: typing.Union[int, float, str]):
        """
        Construct a new instance.

        Parameters
        ----------
        callsign: str
            The Aircraft callsign.
        kind: str
            The kind of action. Currently we support.
                - `on_route`: automatically follow filed Route
                - `set_route`: go directly to named Fix on Route
                - `change_heading_to`: change heading to the specified degrees
                - `change_heading_by`: change heading by given degrees
                - `change_flight_level_to`: go to the specified flight level
                - `change_flight_level_by`: descend/ascend by specified flight levels
        value: Union[int, float, str]
            Allowed values for each Action kind differs:
            - `on_route`: any value is permitted (use 0 as convention)
            - `set_route`: str
            - `change_heading_to`: float
            - `change_heading_by`: float
            - `change_flight_level_to`: int
            - `change_flight_level_by`: int

        Examples
        --------
        >>> action = Action("AIR123", "change_flight_level_to", 300)
        >>> action = Action("AIR123", "set_route", "PORT1")
        >>> action = Action("AIR123", "on_route", 0)
        """

        if len(callsign) == 0:
            raise ValueError("Callsign must not be empty.")

        self.callsign = callsign
        self.kind = kind
        self.value = value

    def from_str(s: str) -> Action:
        """
        Construct a new instance from a string representation.

        Parameters
        ----------
        s: str
            A string representation of Action.

        Returns
        ----------
        Action

        Examples
        ----------
        >>> Action.from_str("NAX123 change_heading_by 5.0")
        """

        s.strip()

        s = s.split()
        callsign = s[0]
        kind = s[1]
        value = " ".join(s[2:])

        return Action(callsign, kind, json.loads(value))


    def __str__(self) -> str:
        """
        Create a human readable representation of the instance.

        Returns
        ----------
        str
            A string representation of Action (e.g., "NAX123 change_heading_by 5.0")
        """

        return f"{self.callsign} {self.kind} {self.value}"
