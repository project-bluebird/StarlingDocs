[docs]    def __init__(self, control_points: list[Pos4D]):
        """
        Construct a new instance.
        """

        if len(control_points) < 2:
            raise ValueError("Trajectory must contain at least two control-points.")

        # stack the (4,) shape arrays into (n, 4) array
        self.locations = np.stack([cp.location for cp in control_points])

        # check that the times are monotonically increasing
        # (difference between them is >= 0)
        if not np.all(np.diff(self.locations[:, 3]) >= 0):
            raise ValueError("Control point times should be sequentially non-decreasing.")

    def from_json(s: str) -> Trajectory:
        """
        Construct a new instance from JSON representation.
        """

        control_points = []
        for cp in json.loads(s):
            control_points.append(Pos4D.from_str(cp))

        return Trajectory(control_points)



[docs]    @staticmethod
    def from_array(a: np.ndarray) -> Trajectory:
        """
        Construct a new instance from a numpy array.
        """

        if a.shape[1] != 4:
            raise ValueError(f"Expected array shape (:,4), given: {a.shape}")

        return Trajectory([Pos4D.from_array(row) for row in a])

    @property
    def control_points(self) -> list[Pos4D]:
        """
        Returns the Trajectory as a list of Pos4D.
        """

        return [Pos4D.from_array(row) for row in self.locations]

[docs]    def min_distance(self, other: Trajectory) -> float:
        """
        Calculate the minimum distance [nmi] to another Trajectory.
        """

        # extract both trajectories (lat,lon,fl) and convert fl to height in nmi
        # note: we copy here to avoid altering the internal state of either trajectory
        t0 = np.copy(self.locations[:, :3])
        t0[:, 2] *= convert.FL_TO_NMI
        t0_times = self.locations[:, 3]

        t1 = np.copy(other.locations[:, :3])
        t1[:, 2] *= convert.FL_TO_NMI
        t1_times = other.locations[:, 3]

        min_dist = np.inf

        # for both trajectories iterate through all consecutive pairs of points and their corresponding times
        for a, b, a_time, b_time in zip(t0[:-1], t0[1:], t0_times[:-1], t0_times[1:]):
            segment_ab = (a, b)

            for c, d, c_time, d_time in zip(t1[:-1], t1[1:], t1_times[:-1], t1_times[1:]):
                segment_cd = (c, d)

                # skip the distance calculation if the start time of a->b occurs after the end time of c->d
                if a_time > d_time:
                    continue

                # stop checking if the start time of c->d occurs after the end time of a->b. we know that all
                # subsequent sections in t1 will occur after the one just evaluated, so there can be no overlap
                elif c_time > b_time:
                    break

                distance = geometry.min_distance_between_segments(segment_ab, segment_cd)

                if distance < min_dist:
                    min_dist = distance

        return min_dist
