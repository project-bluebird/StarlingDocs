[docs]    def __init__(
        self,
        flight_plan: FlightPlan,
        lat: float,
        lon: float,
        fl: float,
        heading: float,
        speed: float,
        vertical_speed: float,
        aircraft_type: typing.Optional[str] = None,
        start_time: float = 0.0,
        requested_heading: typing.Optional[float] = None,
        requested_fl: typing.Optional[int] = None,
    ):
        """
        Construct a new instance.

        Parameters
        ----------
        flight_plan: FlightPlan
            The Aircraft planned Route and the entry/exit Contracts i.e., when and where to 
            enter/exit the Sector.
        lat: float
            The Aircraft latitude in degrees.
        lon: float
            The Aircraft longitude in degrees.
        fl: float
            The Aircraft flight level.
        heading: float
            The Aircraft heading in degrees.
        speed: float
            The Aircraft horizontal speed in knots.
        vertical_speed: float
            The Aircraft vertical speed (climb or descend) [].
        aircraft_type: str, optional
            The Aircraft type (NOTE: currently we do not use this information).
        start_time: float, optional
            The (Environment) time in seconds at which the Aircraft starts travelling on its Route
            (usually a Route starts outside the Sector).
        requested_heading: float, optional
            Agent requested heading in degrees. If None, the Aircraft is following its Route.
        requested_fl: int, optional
            Agent requested flight level (causes the Aircraft to climb/descend).
        """

        if fl < 0.0:
            raise ValueError("Flight level must be non-negative.")

        if speed < 0.0:
            raise ValueError("Aircraft speed must be non-negative.")

        if start_time < 0.0:
            raise ValueError("Start time must be non-negative.")

        self.flight_plan = flight_plan
        self.lat = lat
        self.lon = lon
        self.fl = fl
        self.heading = heading % 360
        self.speed = speed
        self.start_time = start_time
        self.vertical_speed = vertical_speed
        self.aircraft_type = aircraft_type

        if requested_heading is None:
            self.requested_heading = heading % 360
        else:
            self.requested_heading = requested_heading % 360.0
        if requested_fl is None:
            self.requested_fl = fl
        else:
            if requested_fl < 0.0:
                raise ValueError("Requested flight level must be non-negative.")
            self.requested_fl = requested_fl

    def from_json(s: str) -> Aircraft:
        """
        Construct a new instance from a string in JSON format.

        Parameters
        ----------
        s: str
            A string representation of Aircraft in a JSON/dictionary structure.

        Returns
        ----------
        Aircraft

        Examples
        --------
        >>> Aircraft.from_json('''
        >>>  {
        >>>    "flight_plan": {
        >>>        "entry": "160.0FL 00:00:00",
        >>>        "exit": "200.0FL 00:15:00",
        >>>        "route": [
        >>>            "ALFA",
        >>>            "BRAVO",
        >>>            "CHAR"
        >>>        ]
        >>>        },
        >>>    "placement": "51.4702N 0.4479W 120.0FL",
        >>>    "heading": 249.68,
        >>>    "speed": 300,
        >>>    "vertical_speed": 0,
        >>>    "aircraft_type": null,
        >>>    "start_time": 0.0,
        >>>    "requested_heading": 249.68,
        >>>    "requested_fl": 120.0,
        >>>  }''')
        """

        data = json.loads(s)

        flight_plan = FlightPlan.from_json(json.dumps(data["flight_plan"]))

        placement = Pos3D.from_str(f"{data['placement']}")
        lat = placement.lat
        lon = placement.lon
        fl = placement.fl

        heading = data["heading"] % 360.0
        speed = data["speed"]
        vertical_speed = data["vertical_speed"]
        aircraft_type = data.get("aircraft_type", None)
        start_time = data.get("start_time", 0.0)
        requested_heading = data.get("requested_heading", None)
        requested_fl = data.get("requested_fl", None)

        return Aircraft(
            flight_plan,
            lat,
            lon,
            fl,
            heading,
            speed,
            vertical_speed,
            aircraft_type,
            start_time,
            requested_heading,
            requested_fl,
        )

[docs]    def travel(self, trajectory: Trajectory):
        """
        Travel a given Trajectory.

        Parameters
        ----------
        trajectory: Trajectory
            Move the Aircraft position to the last Trajectory point and update
            heading and vertical_speed accorindgly (i.e., in direction of travel).
        """

        if len(trajectory.control_points) == 0:
            return

        final = trajectory.control_points[-1]

        self.lat = final.lat
        self.lon = final.lon
        self.fl = final.fl

        if len(trajectory.control_points) >= 2:
            penultimate = trajectory.control_points[-2]
            self.heading = penultimate.bearing_to(final) % 360.0
            self.vertical_speed = final.fl - penultimate.fl / (final.time - penultimate.time)



[docs]    def distance(self, other: Pos3D) -> float:
        """
        Calculate lateral distance [nmi] between Aircraft and another Pos3D location
        (vertical distance is ignored).

        Parameters
        ----------
        other: Pos3D
            A latitude, longitude, altitude position.

        Returns
        ----------
        float
            Lateral distance in nautical miles between Aircraft and the given 3D position.
        """

        return self.pos3d().distance(other)



[docs]    def distance_to_exit(self, airspace: Airspace) -> float:
        """
        Calculate Aircraft lateral distance [nmi] to the contracted exit Pos3D.

        Returns
        ----------
        float
            Lateral distance in nautical miles to the exit Fix.
        """

        exit_2d = airspace.route_exit_fix(self.flight_plan.route)
        exit_3d = exit_2d.pos3d(self.flight_plan.exit.fl)
        return self.distance(exit_3d)

    def set_bada_parameters(self) -> None:

        if self.aircraft_type is not None:
            try:
                with open("resources/json/aircraft_database.json") as aircraft_database:
                    self.bada_parameters = json.load(aircraft_database)["aircrafts"][self.aircraft_type]
            except FileNotFoundError:
                warnings.warn(" Aircraft database not found. Using default aircraft parameters... \n")
                self.bada_parameters = None
            except KeyError:
                warnings.warn(
                    ' Aircraft type "{}" not found in database. Using default aircraft parameters... \n'.format(
                        self.aircraft_type
                    )
                )
                self.bada_parameters = None
        else:
            self.bada_parameters = None
