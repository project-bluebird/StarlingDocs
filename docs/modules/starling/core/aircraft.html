[docs]    def __init__(
        self,
        flight_plan: FlightPlan,
        lat: float,
        lon: float,
        fl: float,
        heading: float,
        speed: float,
        vertical_speed: float,
        aircraft_type: typing.Optional[str] = None,
        start_time: float = 0.0,
        requested_heading: typing.Optional[float] = None,
        requested_fl: typing.Optional[float] = None
    ):
        """
        Construct a new instance.
        """

        if fl < 0.0:
            raise ValueError("Flight level must be non-negative.")

        if speed < 0.0:
            raise ValueError("Aircraft speed must be non-negative.")

        if start_time < 0.0:
            raise ValueError("Start time must be non-negative.")

        self.flight_plan = flight_plan
        self.lat = lat
        self.lon = lon
        self.fl = fl
        self.heading = heading % 360
        self.speed = speed
        self.start_time = start_time
        self.vertical_speed = vertical_speed
        self.aircraft_type = aircraft_type

        if requested_heading is None:
            self.requested_heading = heading % 360
        else:
            self.requested_heading = requested_heading % 360.0
        if requested_fl is None:
            self.requested_fl = fl
        else:
            if requested_fl < 0.0:
                raise ValueError("Requested flight level must be non-negative.")
            self.requested_fl = requested_fl

    def from_json(s: str) -> Aircraft:
        """
        Construct a new instance from JSON representation.
        """

        data = json.loads(s)

        flight_plan = FlightPlan.from_json(json.dumps(data["flight_plan"]))

        placement = Pos3D.from_str(f"{data['placement']}")
        lat = placement.lat
        lon = placement.lon
        fl = placement.fl

        heading = data["heading"] % 360.0
        speed = data["speed"]
        vertical_speed = data["vertical_speed"]
        aircraft_type = data.get("aircraft_type", None)
        start_time = data.get("start_time", 0.0)
        requested_heading = data.get("requested_heading", None)
        requested_fl = data.get("requested_fl", None)

        return Aircraft(
            flight_plan,
            lat,
            lon,
            fl,
            heading,
            speed,
            vertical_speed,
            aircraft_type,
            start_time,
            requested_heading,
            requested_fl,
        )

[docs]    def travel(self, trajectory: Trajectory):
        """
        Travel a given Trajectory.
        """

        if len(trajectory.control_points) == 0:
            return

        final = trajectory.control_points[-1]

        self.lat = final.lat
        self.lon = final.lon
        self.fl = final.fl

        if len(trajectory.control_points) >= 2:
            penultimate = trajectory.control_points[-2]
            self.heading = penultimate.bearing_to(final) % 360.0
            self.vertical_speed = final.fl - penultimate.fl / (final.time - penultimate.time)

[docs]    def distance_to_exit(self, airspace: Airspace) -> float:
        """
        Calculate Aircraft distance [nmi] to the contracted exit Pos3D.
        """

        exit_2d = airspace.route_exit_fix(self.flight_plan.route)
        exit_3d = exit_2d.pos3d(self.flight_plan.exit.fl)
        return self.distance(exit_3d)
