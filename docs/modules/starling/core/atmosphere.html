    """
    Class to describe the atmospheric conditions in a BADA atmospheric model.
    See EEC Technical/Scientific Report No. 2010-001 for specific details.
    If you instantiate this class without any arguments it will return by
    default an ISA (International Standard Atmosphere). Note below MSL refers
    to mean sea level.
    """


[docs]    def __init__(self, delta_T: float = 0.0, delta_P: float = 0.0):
        """
        Construct a new instance of an atmosphere

        Parameters
        ----------
        delta_T : float = 0.0
            The BADA atmosphere temperature differential at a pressure
            altitude of zero (in K)
        delta_P : float = 0.0
            The BADA atmosphere pressure differential at a pressure
            altitude of zero (in Pa)
        """
        if delta_P != 0.0:
            P_msl = constants.P_ZERO + delta_P
            expo = -1.0 * constants.BETA_T * constants.R_GAS / constants.G_ACC
            Hp_msl = (constants.T_ZERO / constants.BETA_T) * ((P_msl / constants.P_ZERO) ** expo - 1.0)
        else:
            P_msl = constants.P_ZERO
            Hp_msl = 0.0

        T_isa_msl = constants.T_ZERO + constants.BETA_T * Hp_msl
        T_msl = T_isa_msl + delta_T

        T_isa_trop = constants.T_ZERO + constants.BETA_T * constants.HP_TROP
        T_trop = T_isa_trop + delta_T

        self.delta_T = delta_T  # Temperature differential at MSL
        self.delta_P = delta_P  # Pressure differential at MSL
        self.P_msl = P_msl  # Pressure at MSL
        self.Hp_msl = Hp_msl  # Pressure height at MSL
        self.T_isa_msl = T_isa_msl  # Temp. at MSL assuming an ISA
        self.T_msl = T_msl  # Temp. at MSL in actual BADA atmosphere
        self.T_isa_trop = T_isa_trop  # Temp. at MSL assuming an ISA
        self.T_trop = T_trop  # Temp. at tropopause



[docs]    def temperature_from_fl(self, flight_level: float) -> float:
        """
        Calculate the temperature in an atmosphere at a given flight level.
        Note here we first convert the flight level to a pressure altitude (Hp)
        See equations 3.28 in EEC Technical/Scientific Report No. 2010-001.

        Parameters
        ----------
        flight_level : float
            A flight level

        Returns
        -------
        float
            The bada atmospheric temperature at the corresponding flight level
            height.
        """
        Hp = flight_level * convert.FL_TO_M
        if Hp <= constants.HP_TROP:
            temp = constants.T_ZERO + self.delta_T + constants.BETA_T * Hp
        else:
            temp = self.T_trop
        return temp



[docs]    def pressure_from_fl(self, flight_level: float) -> float:
        """
        Calculate the pressure in an atmosphere at a given flight level.
        Note here we first convert the flight level to a pressure altitude (Hp)
        See equations 3.26 in EEC Technical/Scientific Report No. 2010-001.

        Parameters
        ----------
        flight_level : float
            A flight level

        Returns
        -------
        float
            The bada atmospheric pressure at the corresponding flight level
            height.
        """
        Hp = flight_level * convert.FL_TO_M
        expo = -1.0 * constants.G_ACC / (constants.BETA_T * constants.R_GAS)
        if Hp <= constants.HP_TROP:
            press = constants.P_ZERO * (1.0 + constants.BETA_T * Hp / constants.T_ZERO) ** expo
        else:
            P_trop = constants.P_ZERO * (1.0 + constants.BETA_T * constants.HP_TROP / constants.T_ZERO) ** expo
            press = P_trop * np.exp(
                -1.0 * constants.G_ACC * (Hp - constants.HP_TROP) / (constants.R_GAS * self.T_isa_trop)
            )
        return press



[docs]    def density_from_fl(self, flight_level: float) -> float:
        """
        Calculate the density in an atmosphere at a given flight level.

        Parameters
        ----------
        flight_level : float
            A flight level

        Returns
        -------
        float
            The bada atmospheric density at the corresponding flight level
            height.
        """
        press = self.pressure_from_fl(flight_level)
        temp = self.temperature_from_fl(flight_level)
        density = press / (constants.R_GAS * temp)
        return density



[docs]    def geopot_altitude_from_fl(self, flight_level: float) -> float:
        """
        Calculate the geopotential height at a given flight level, by
        first converting the flight level to pressure altitude (Hp) and
        the using equations 3-31 in EEC Technical/Scientific Report
        No. 2010-001.

        Parameters
        ----------
        flight_level : float
            A flight level

        Returns
        -------
        float
            The geopotential height in the bada atmosphere at the given flight
            level height.
        """
        Hp = flight_level * convert.FL_TO_M
        if Hp <= constants.HP_TROP:
            altitude = (
                Hp
                - self.Hp_msl
                + (self.delta_T / constants.BETA_T)
                * np.log((constants.T_ZERO + constants.BETA_T * Hp) / self.T_isa_msl)
            )
        else:
            altitude = constants.HP_TROP + (self.T_trop / self.T_isa_trop) * (Hp - constants.HP_TROP)
        return altitude



[docs]    def geodet_altitude_from_fl(self, flight_level: float) -> float:
        """
        Calculate the geodetic height at a given flight level. Here we
        first calculate the geopotential height and then convert to
        geodetic height using equation 2-7 in EEC Technical/Scientific Report
        No. 2010-001.

        Parameters
        ----------
        flight_level : float
            A flight level

        Returns
        -------
        float
            The geodetic height in the bada atmosphere at the given flight level
            height.
        """
        H = self.geopot_altitude_from_fl(flight_level)
        altitude = constants.R_E * H / (constants.R_E - H)
        return altitude



[docs]    def cas_to_tas(self, flight_level: float, calibrated_air_speed: float) -> float:
        """
        Convert calibrated air speed (CAS) to true air speed (TAS).
        See equation 3.1-23 in EEC Technical/Scientific Report No. 2010-003.

        Parameters
        ----------
        flight_level : float
            A flight level
        calibrated_air_speed : float
            The calibrated air speed of the aircraft (in m/s)

        Returns
        -------
        float
            The true air speed of the aircraft (in m/s)
        """
        mu = (constants.KAPPA - 1.0) / constants.KAPPA
        p_over_rho = self.pressure_from_fl(flight_level) / self.density_from_fl(flight_level)
        p_over_rho_zero = constants.P_ZERO / constants.RHO_ZERO
        p_ratio = constants.P_ZERO / self.pressure_from_fl(flight_level)

        x = 1.0 + 0.5 * mu * (calibrated_air_speed**2) / p_over_rho_zero
        x = (1.0 + p_ratio * (x ** (1.0 / mu) - 1.0)) ** mu
        true_air_speed = np.sqrt(2.0 * p_over_rho * (x - 1.0) / mu)
        return true_air_speed



[docs]    def tas_to_cas(self, flight_level: float, true_air_speed: float) -> float:
        """
        Convert true air speed (TAS) to calibrated air speed (CAS).
        See equation 3.1-24 in EEC Technical/Scientific Report No. 2010-003.

        Parameters
        ----------
        flight_level : float
            A flight level
        true_air_speed : float
            The true air speed of the aircraft (in m/s)

        Returns
        -------
        float
            The calibrated air speed of the aircraft (in m/s)
        """
        mu = (constants.KAPPA - 1.0) / constants.KAPPA
        p_over_rho = self.pressure_from_fl(flight_level) / self.density_from_fl(flight_level)
        p_over_rho_zero = constants.P_ZERO / constants.RHO_ZERO
        p_ratio = self.pressure_from_fl(flight_level) / constants.P_ZERO

        x = 1.0 + 0.5 * mu * (true_air_speed**2) / p_over_rho
        x = (1.0 + p_ratio * (x ** (1.0 / mu) - 1.0)) ** mu
        calibrated_air_speed = np.sqrt(2.0 * p_over_rho_zero * (x - 1.0) / mu)
        return calibrated_air_speed

        true_air_speed = ground_speed
        return true_air_speed

[docs]    def tas_to_mach(self, flight_level: float, true_air_speed: float) -> float:
        """
        Converts the true airspeed (TAS) to a mach number.
        See equation 3.1-26 in EEC Technical/Scientific Report No. 2010-003.

        Parameters
        ----------
        flight_level : float
            A flight level
        true_air_speed : float
            The true air speed of the aircraft (in m/s)

        Returns
        -------
        float
            The aircrafts speed as a mach number (dimensionless)
        """
        mach = true_air_speed / np.sqrt(constants.KAPPA * constants.R_GAS * self.temperature_from_fl(flight_level))
        return mach

[docs]    def mach_cas_trans_altitude(self, mach: float, calibrated_air_speed: float) -> float:
        """
        For given mach number and calibrated air speed find the pressure altitude at which
        these two quantities represent the same true air speed.
        See equations 3.1-27 to 3.1-29 in EEC Technical/Scientific Report No. 2010-003.

        Parameters
        ----------
        mach : float
            The given mach number
        calibrated_air_speed : float
            The given calibrated air speed (in m/s)

        Returns
        -------
        float
            The transition pressure altitude (in m)
        """
        mu_inv = constants.KAPPA / (constants.KAPPA - 1.0)
        kappa_prime = (constants.KAPPA - 1.0) / 2.0
        pressure_ratio = ((1.0 + kappa_prime * (calibrated_air_speed / constants.A_ZERO) ** 2) ** mu_inv - 1.0) / (
            (1.0 + kappa_prime * mach**2) ** mu_inv - 1.0
        )
        temperature_ratio = pressure_ratio ** (-1.0 * constants.BETA_T * constants.R_GAS / constants.G_ACC)

        transition_altitude = (
            (1.0 - temperature_ratio) * (constants.T_ZERO + self.delta_T) * (1000.0 / (0.304 * 6.5)) * convert.FT_TO_M
        )
        return transition_altitude
