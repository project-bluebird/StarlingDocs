    def from_json(s: str) -> Plan:
        """
        Construct a new instance from JSON representation.
        """

        positions = []
        for pos in json.loads(s):
            positions.append(Pos3D.from_str(pos))

        return Plan(positions)

[docs]    def min_distance(self, other: Plan) -> float:
        """
        Calculate the minimum distance [nmi] to another Plan.
        """

        from ..utility import convert, geometry

        min = self.positions[0].distance(other.positions[0])
        for segment0 in self.segments():
            (a, b) = segment0
            s0 = [[a.lat, a.lon, a.fl * convert.FL_TO_NMI], [b.lat, b.lon, b.fl * convert.FL_TO_NMI]]

            for segment1 in other.segments():
                (c, d) = segment1
                s1 = [[c.lat, c.lon, c.fl * convert.FL_TO_NMI], [d.lat, d.lon, d.fl * convert.FL_TO_NMI]]

                d = geometry.min_distance_between_segments(s0, s1)
                if d < min:
                    min = d

        return min
