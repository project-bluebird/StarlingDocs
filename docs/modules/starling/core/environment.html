[docs]    def __init__(self, time: float, airspace: Airspace, aircraft: dict[str, Aircraft]):
        """
        Construct a new instance.
        """

        if time < 0.0:
            raise ValueError("Time must be non-negative.")

        self.time = time
        self.airspace = airspace
        self.aircraft = aircraft
        self.on_route = {}
        for callsign in self.aircraft.keys():
            self.on_route[callsign] = True



[docs]    @staticmethod
    def from_json(s: str) -> Environment:
        """
        Construct a new instance from JSON representation.
        """

        data = json.loads(s)

        time = convert.time_str(data["time"])

        airspace = Airspace.from_json(json.dumps(data["airspace"]))

        aircraft = {}
        for callsign, route in data["aircraft"].items():
            aircraft[callsign] = Aircraft.from_json(json.dumps(route))

        return Environment(time, airspace, aircraft)

[docs]    def data(self) -> dict:
        """
        Get the data as a serialisable dictionary.
        """

        timestamp = time.gmtime(self.time)
        time_str = time.strftime("%H:%M:%S", timestamp)

        return {
            "time": time_str,
            "airspace": self.airspace.data(),
            "aircraft": dict(list(map(lambda pair: (pair[0], pair[1].data()), self.aircraft.items()))),
        }

[docs]    def is_active(self) -> dict:
        """
        Determine which Aircraft are active within the Airspace.
        """

        active = {}

        for callsign, aircraft in self.aircraft.items():
            active[callsign] = self.airspace.contains(aircraft.pos3d())

        return active



[docs]    def exited(self) -> dict:
        """
        Determine which Aircraft have exited the Airspace.
        """

        exited = {}

        for callsign, aircraft in self.aircraft.items():
            if self.time > aircraft.flight_plan.entry.time and not self.airspace.contains(aircraft.pos3d()):
                exited[callsign] = True
            else:
                exited[callsign] = False

        return exited



[docs]    def finished(self) -> bool:
        """
        Determine if all Aircraft have exited the Airspace i.e. Scenario is finished.
        """

        return all(list(self.exited().values()))



[docs]    def aircraft_lateral_distance(self, aircraft_list):
        """
        Compute distance matrix using lats/lons of aircraft
        """

        if len(aircraft_list) < 2:
            raise ValueError("len(aircraft_list) must be >= 2")

        # Collect latitudes/longitudes of active aircraft
        lats = []
        lons = []
        for ac in aircraft_list:
            lats.append(self.aircraft[ac].lat * convert.DEG_TO_NMI)
            lons.append(self.aircraft[ac].lon * convert.DEG_TO_NMI)

        dist_matrix = geometry.compute_distance_matrix(np.vstack([np.array(lats), np.array(lons)]).T)
        return dist_matrix



[docs]    def active_aircraft_lateral_distance(self):
        """
        A convenient function to first determine which Aircraft are active within the Airspace
            and then compute distance matrix using lats/lons
        It returns both the list of active aircraft and distance matrix
        """

        # List of active aircrafts
        aircraft = self.is_active()

        def _by(x):
            return aircraft[x]

        aircraft_list = list(filter(_by, aircraft))

        if len(aircraft_list) > 1:
            # Compute distance matrix
            dist_matrix = self.aircraft_lateral_distance(aircraft_list)
        else:
            dist_matrix = None

        return aircraft_list, dist_matrix



[docs]    def extract_exterior_coords(self):
        """
        Extract coordinates of the exterior boundary
        """
        points = []
        for sector in self.airspace.sectors.values():
            for volume in sector.volumes:
                for coord in volume.area.boundary.exterior.coords:
                    points.append((coord[0], coord[1]))
        return zip(*points)



[docs]    def extract_fixes(self):
        """
        Extract fixes
        """
        x_f = []
        y_f = []
        n_f = []
        for name, location in self.airspace.fixes.places.items():
            x_f.append(location.lon)
            y_f.append(location.lat)
            n_f.append(name)
        return x_f, y_f, n_f



[docs]    def distances_to_exit(self) -> dict:
        """
        Return distance [nmi] to exit for all active Aircraft.
        """

        distances = {}
        aircraft = self.is_active()
        for callsign, active in aircraft.items():
            if active:
                a = self.aircraft[callsign]
                distances[callsign] = a.distance_to_exit(self.airspace.fixes)
        return distances
