[docs]    def __init__(self, area: Area, min_fl: float, max_fl: float):
        """
        Construct a new instance.
        """

        if min_fl < 0.0:
            raise ValueError("Minimum flight level must be non-negative.")

        if max_fl <= min_fl:
            raise ValueError("Maximum flight level must be greater than minimum flight level.")

        self.area = area
        self.min_fl = min_fl
        self.max_fl = max_fl



[docs]    @staticmethod
    def from_json(s: str) -> Volume:
        """
        Construct a new instance from JSON representation.
        """

        data = json.loads(s)

        area = Area.from_json(json.dumps(data["area"]))
        min_fl = float(data["min_fl"])
        max_fl = float(data["max_fl"])

        return Volume(area, min_fl, max_fl)

[docs]    def data(self) -> dict:
        """
        Get the data as a serialisable dictionary.
        """

        return {
            "area": list(map(lambda p: f"{Pos2D(p[1], p[0])}", self.area.boundary.exterior.coords))[:-1],
            "min_fl": self.min_fl,
            "max_fl": self.max_fl,
        }

[docs]    def contains(self, position: Pos3D) -> bool:
        """
        Determine if a given position is contained within the Volume.
        """

        if position.fl < self.min_fl or self.max_fl < position.fl:
            return False

        return self.area.contains(position.pos2d())
