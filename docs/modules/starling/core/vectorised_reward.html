    Comprises a vector of values, one for each time-step, calculated by a VectorisedObjective.
    """

        """
        Construct a new instance.

        Parameters
        ----------
        values : np.ndarray
            A vector of values calculated by a VectorisedObjective
        **extra_data
            Arbitrary keyword arguments (kwargs)

        Examples
        --------
        >>> reward = Reward(np.array([0.6, 0.3, 0.7]), a="double", b="checking")

        if not isinstance(values, np.ndarray):
            raise TypeError("A vector of type np.ndarray must be provided.")

        if np.any(np.isnan(values)):
            raise ValueError("Reward values must not be NAN.")

        self.values = values

        for attribute, x in extra_data.items():
            setattr(self, attribute, x)
