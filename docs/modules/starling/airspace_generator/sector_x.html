[docs]    def __init__(self, width: float, height: float, fl_limits: typing.Tuple[float, float], alpha: float):
        """
        Construct a new instance.

        Parameters
        ----------
        width: float
            The width of the airways.
        height: float
           The height of the Airspace (includes exterior Fixes).
        fl_limits: list[float]
            The [min, max] flight level limits allowed in the Sector.
        alpha: float
            The absolute degrees by which both branches are rotated from a vertical line (0 < alpha < 90).
            For example, alpha=45 results in an X shape with 90 degrees between all branches.
        """

        if height <= 0:
            raise ValueError("Height must be positive.")

        if (fl_limits[0] < 0) or (fl_limits[1] < 0):
            raise ValueError("Flight level limits must be positive.")

        if fl_limits[0] >= fl_limits[1]:
            raise ValueError("Maximum flight level must exceed minimum flight level.")

        if alpha <= 0.0 or alpha >= 90:
            raise ValueError("Alpha must be positive in the range (0,90).")

        self.width = width * convert.NMI_TO_DEG
        self.height = height * convert.NMI_TO_DEG
        self.fl_limits = fl_limits
        self.alpha = alpha



[docs]    def generate_airspace(self) -> typing.Tuple[Airspace, typing.List[Route]]:
        """
        Generate an airspace.

        Returns
        ----------
        Tuple[Airspace, list[Route]]
            A tuple with (Airspace, list of Routes).
        """

        [min_fl, max_fl] = self.fl_limits

        north_west = Pos2D(self.height * 0.4, -self.width * 0.5)
        south_west = Pos2D(-self.height * 0.4, -self.width * 0.5)
        south_east = Pos2D(-self.height * 0.4, self.width * 0.5)
        north_east = Pos2D(self.height * 0.4, self.width * 0.5)

        b = geometry.rotate(north_east, self.alpha)
        c = geometry.rotate(north_west, self.alpha)
        e = geometry.rotate(south_west, -self.alpha)
        f = geometry.rotate(south_east, -self.alpha)
        h = geometry.rotate(south_west, self.alpha)
        i = geometry.rotate(south_east, self.alpha)
        k = geometry.rotate(north_east, -self.alpha)
        l = geometry.rotate(north_west, -self.alpha)

        a = geometry.intersection((b, i), (Pos2D(self.height, 0), Pos2D(-self.height, 0)))
        d = geometry.intersection((c, h), (Pos2D(0, -self.height), Pos2D(0, self.height)))
        j = geometry.intersection((f, k), (Pos2D(0, -self.height), Pos2D(0, self.height)))
        g = geometry.intersection((c, h), (Pos2D(self.height, 0), Pos2D(-self.height, 0)))

        # create the outer boundary of the sector

        # use the points in the inner boundary to define 4 lines (all segments of the inner boundary)
        line_north_west = np.array([[b.lon, b.lat], [c.lon, c.lat]])
        line_south_west = np.array([[e.lon, e.lat], [f.lon, f.lat]])
        line_south_east = np.array([[h.lon, h.lat], [i.lon, i.lat]])
        line_north_east = np.array([[k.lon, k.lat], [l.lon, l.lat]])

        # solve the linear equations to find intersecting points between lines
        west_x, _ = np.linalg.solve(np.array([line_north_west[1] - line_north_west[0],
                                         line_south_west[0] - line_south_west[1]]).T,
                               line_south_west[0] - line_north_west[0])
        south_x, _ = np.linalg.solve(np.array([line_south_west[1] - line_south_west[0],
                                                               line_south_east[0] - line_south_east[1]]).T,
                                                     line_south_east[0] - line_south_west[0])
        east_x, _ = np.linalg.solve(np.array([line_south_east[1] - line_south_east[0],
                                                                 line_north_east[0] - line_north_east[1]]).T,
                                                       line_north_east[0] - line_south_east[0])
        north_x, _ = np.linalg.solve(np.array([line_north_east[1] - line_north_east[0],
                                                               line_north_west[0] - line_north_west[1]]).T,
                                                     line_north_west[0] - line_north_east[0])

        point_west_outer = (1 - west_x) * line_north_west[0] + west_x * line_north_west[1]
        point_south_outer = (1 - south_x) * line_south_west[0] + south_x * line_south_west[1]
        point_east_outer = (1 - east_x) * line_south_east[0] + east_x * line_south_east[1]
        point_north_outer = (1 - north_x) * line_north_east[0] + north_x * line_north_east[1]

        # use the intersecting points to define an Area that contains a shapely Polygon defining the outer boundary
        outer_boundary_points = [point_west_outer, point_south_outer, point_east_outer, point_north_outer]
        outer_boundary = Area([Pos2D(point[1], point[0]) for point in outer_boundary_points])

        sectors = {"sector_x": Sector([Volume(Area([a, b, c, d, e, f, g, h, i, j, k, l]), min_fl, max_fl)])}

        fixes = Fixes(
            {
                "ABYSS": Pos2D(0, 0),
                "DEMON": geometry.rotate(Pos2D(self.height * 0.4, 0), -self.alpha),
                "SANTA": geometry.rotate(Pos2D(self.height * 0.5, 0), -self.alpha),
                "GATES": geometry.rotate(Pos2D(self.height * 0.4, 0), self.alpha),
                "SIN": geometry.rotate(Pos2D(self.height * 0.5, 0), self.alpha),
                "WITCH": geometry.rotate(Pos2D(-self.height * 0.4, 0), -self.alpha),
                "SIREN": geometry.rotate(Pos2D(-self.height * 0.5, 0), -self.alpha),
                "HAUNT": geometry.rotate(Pos2D(-self.height * 0.4, 0), self.alpha),
                "LIMBO": geometry.rotate(Pos2D(-self.height * 0.5, 0), self.alpha),
            }
        )

        routes = [
            Route(["SIREN", "WITCH", "ABYSS", "DEMON", "SANTA"]),
            Route(["LIMBO", "HAUNT", "ABYSS", "GATES", "SIN"]),
            Route(["SIN", "GATES", "ABYSS", "HAUNT", "LIMBO"]),
            Route(["SANTA", "DEMON", "ABYSS", "WITCH", "SIREN"]),
        ]

        return (Airspace(sectors, fixes, outer_boundary), routes)
