[docs]    def __init__(self, width: float, height: float, fl_limits: typing.Tuple[float, float], alpha: float):
        """
        Construct a new instance.

        Parameters
        ----------
        width: float
            The width of the airways.
        height: float
           The height of the Airspace (includes exterior Fixes).
        fl_limits: list[float]
            The [min, max] flight level limits allowed in the Sector.
        alpha: float
            The absolute degrees by which both branches are rotated from a vertical line (0 < alpha < 180).
            For example, alpha=90 results in a T shaped Sector.
        """

        if width <= 0:
            raise ValueError("Width must be positive.")

        if height <= 0:
            raise ValueError("Height must be positive.")

        if (fl_limits[0] < 0) or (fl_limits[1] < 0):
            raise ValueError("Flight level limits must be positive.")

        if fl_limits[0] >= fl_limits[1]:
            raise ValueError("Maximum flight level must exceed minimum flight level.")

        if alpha <= 0.0 or alpha >= 180:
            raise ValueError("Alpha must be positive in the range(0,180).")

        self.width = width * convert.NMI_TO_DEG
        self.height = height * convert.NMI_TO_DEG
        self.fl_limits = fl_limits
        self.alpha = alpha



[docs]    def generate_airspace(self) -> typing.Tuple[Airspace, typing.List[Route]]:
        """
        Generate an airspace.
        """

        [min_fl, max_fl] = self.fl_limits

        north_west = Pos2D(self.height * 0.4, -self.width * 0.5)
        south_west = Pos2D(-self.height * 0.4, -self.width * 0.5)
        south_east = Pos2D(-self.height * 0.4, self.width * 0.5)
        north_east = Pos2D(self.height * 0.4, self.width * 0.5)

        b = geometry.rotate(north_east, self.alpha)
        c = geometry.rotate(north_west, self.alpha)
        e = south_west
        f = south_east
        h = geometry.rotate(north_east, -self.alpha)
        i = geometry.rotate(north_west, -self.alpha)

        a = geometry.intersection(
            (b, geometry.rotate(south_east, self.alpha)), (Pos2D(-self.height, 0), Pos2D(self.height, 0))
        )
        d = geometry.intersection((e, north_west), (c, geometry.rotate(e, self.alpha)))
        g = geometry.intersection((f, north_east), (h, geometry.rotate(f, -self.alpha)))

        sectors = {"sector_y": Sector([Volume(Area([a, b, c, d, e, f, g, h, i]), min_fl, max_fl)])}

        fixes = Fixes(
            {
                "GOD": geometry.rotate(Pos2D(self.height * 0.5, 0), -self.alpha),
                "CANON": geometry.rotate(Pos2D(self.height * 0.4, 0), -self.alpha),
                "GHOST": geometry.rotate(Pos2D(self.height * 0.5, 0), self.alpha),
                "BISHP": geometry.rotate(Pos2D(self.height * 0.4, 0), self.alpha),
                "TRI": Pos2D(0, 0),
                "SON": Pos2D(-self.height * 0.4, 0),
                "DECAN": Pos2D(-self.height * 0.5, 0),
            }
        )

        routes = [
            Route(["GOD", "CANON", "TRI", "SON", "DECAN"]),
            Route(["DECAN", "SON", "TRI", "CANON", "GOD"]),
            Route(["GHOST", "BISHP", "TRI", "SON", "DECAN"]),
            Route(["DECAN", "SON", "TRI", "BISHP", "GHOST"]),
        ]

        return (Airspace(sectors, fixes), routes)
