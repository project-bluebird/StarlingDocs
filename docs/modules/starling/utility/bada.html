DEFAULT_ROCD_IN_FL_PER_MINUTE = 20.0



[docs]def change_in_pressure_height_and_mass(
    aircraft: Aircraft, atmos: Atmosphere, aircraft_state: str, time_interval: float
) -> typing.Tuple[float]:
    """
    For a given aircraft find its change in pressure height and mass
    over a given time interval using a fourth order Runge-Kutta integration.

    Parameters
    ----------
    aircraft : Aircraft
        The aircraft for which to find the change in pressure height and mass
    atmos : Atmosphere
        The BADA atmosphere used to represent the atmosphere at lat and lon of
        the aircraft.
    aircraft_state : str
        The state of the aircraft.
    time_interval : float
        The time interval in seconds

    Returns
    -------
    Tuple
        Two floats which are the change in pressure height (in meters) and
        the mass (in kg) at end of time-step
    """

    y_0 = (aircraft.fl * convert.FL_TO_M, aircraft.operation_params["mass"])
    solution = solve_ivp(
        lambda t, y: bada_dy_dt(t, y, atmos, aircraft, aircraft_state), (0, time_interval), np.array(y_0)
    )

    return solution.y[0, -1] - y_0[0], solution.y[1, -1] - y_0[1]




[docs]def bada_dy_dt(
    time: float, y: typing.Tuple[float], atmos: Atmosphere, aircraft: Aircraft, aircraft_state: str
) -> typing.Tuple[float]:
    """
    For a given aircraft find its rate of climb or descent (ROCD i.e. dHp/dt)
    and its rate of mass change.
    Here we use equation 3.2-7 (with the reduced climb power correction, see
    equation 3.8-2) in EEC Technical/Scientific Report No. 2010-003.
    Note if the BADA parameters are not set in the aircraft operation_params
    dictionary, a default ROCD of 20 fl/minute are used with not change in
    mass.

    Parameters
    ----------
    time : float
        Dummy time variable for solve_ivp algorithm
    pressure_height : Tuple
        Tuple of two floats containing the pressure_height level of the aircraft
        and its mass in that order.
    atmos : Atmosphere
        The BADA atmosphere used to represent the atmosphere at lat and lon of
        the aircraft.
    aircraft : Aircraft
        An Aircraft instance.
    aircraft_state : str
        The state of the aircraft.

    Returns
    -------
    Tuple
        Tuple of two floats containting the rate of climb or descent (in m/s),
        and the rate of mass change (kg/s) in that order.
    """

    pressure_height = y[0]
    mass = y[1]

    if "bada" not in aircraft.operation_params:
        if aircraft_state == "cruise":
            rocd = 0.0
        elif aircraft_state == "climb":
            rocd = DEFAULT_ROCD_IN_FL_PER_MINUTE * convert.FL_TO_M / 60.0
        else:
            # Aircraft is either in descent, approach or landing state
            rocd = -1.0 * DEFAULT_ROCD_IN_FL_PER_MINUTE * convert.FL_TO_M / 60.0
        romc = 0.0
    else:
        bada_parameters = aircraft.operation_params["bada"]
        true_air_speed = atmos.gs_to_tas(aircraft.speed)
        temperature = atmos.temperature_from_fl(pressure_height * convert.M_TO_FL)
        density = atmos.density_from_fl(pressure_height * convert.M_TO_FL)
        mach = atmos.tas_to_mach(pressure_height * convert.M_TO_FL, true_air_speed)
        delta_t = atmos.delta_T

        # Calculate thrust and drag values. Note the the returned thrust values for thrust during
        # cruise and climb phases are the maximum values.
        banking_angle = get_banking_angle()
        drag = get_drag(bada_parameters, mass, true_air_speed, density, banking_angle, aircraft_state)
        thrust = get_thrust(bada_parameters, pressure_height, true_air_speed, delta_t, aircraft_state)
        if aircraft_state == "cruise":
            # In this case the thrust must equal the drag by definition
            thrust = drag
        if aircraft_state == "climb":
            c_pow_red = reduced_climb_power_correction(bada_parameters, pressure_height, mass)
        else:
            c_pow_red = 1.0

        if aircraft_state == "climb" and thrust < drag:
            # It sometimes happens the drag calculated is more than the maximum thrust. This
            # probably means the supplied speed is too large. Print warning and set climb rate
            # to the default value:
            warnings.warn("Bada thrust is less than drag during climb. Setting climb rate to default value.")
            rocd = DEFAULT_ROCD_IN_FL_PER_MINUTE
        else:
            rocd = (
                (1.0 - delta_t / temperature)
                * ((thrust - drag) * c_pow_red * true_air_speed / (mass * constants.G_ACC))
                * energy_share_factor(
                    flight_condition(bada_parameters, atmos, aircraft_state, pressure_height),
                    mach,
                    temperature,
                    delta_t,
                )
            )
        romc = rate_of_mass_change(bada_parameters, pressure_height, true_air_speed, thrust, mass, aircraft_state)
    return rocd, romc




[docs]def flight_condition(bada_parameters: dict, atmos: Atmosphere, aircraft_state: str, pressure_height: float) -> str:
    """
    Returns the aircraft flight condition

    Parameters
    ----------
    bada_parameters : dict
        The aircraft parameters from the BADA database.
    atmos : Atmosphere
        The BADA atmosphere used to represent the atmosphere at lat and lon of
        the aircraft.
    aircraft_state : str
        The state of the aircraft.
    pressure_height : float
        The pressure height of the aircraft (in m)

    Returns
    -------
    str:
        The flight condition of the aircraft
    """
    if aircraft_state == "climb":
        cas_hi = bada_parameters["CAScl2"][0] * convert.KT_TO_MPS
        mach = bada_parameters["Mcl"][0]
    elif aircraft_state == "cruise":
        cas_hi = bada_parameters["CAScr2"][0] * convert.KT_TO_MPS
        mach = bada_parameters["Mcr"][0]
    else:
        cas_hi = bada_parameters["CASdes2"][0] * convert.KT_TO_MPS
        mach = bada_parameters["Mdes"][0]
    mach_transition_altitude = atmos.mach_cas_trans_altitude(mach=mach, calibrated_air_speed=cas_hi)
    if pressure_height < mach_transition_altitude:
        if pressure_height < constants.HP_TROP:
            flight_condition = "constant cas below tropopause"
        else:
            flight_condition = "constant cas above tropopause"
    else:
        if pressure_height < constants.HP_TROP:
            flight_condition = "constant mach below tropopause"
        else:
            flight_condition = "constant mach above tropopause"
    return flight_condition




[docs]def rate_of_mass_change(
    bada_parameters: dict,
    pressure_height: float,
    true_air_speed: float,
    thrust: float,
    mass: float,
    aircraft_state: str,
) -> float:
    """
    Returns the aircraft's rate of mass change. The latter is taken to be entirely
    due to fuel consumption. The relevant equations are 3.9-1 to 3.9-9 in EEC
    Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    bada_parameters : dict or None
        The aircraft parameters from the BADA database.
    pressure_height : float
        The pressure_height level of the aircraft
    true_air_speed : float
        The true air speed of the aircraft (in m/s)
    thrust : float
        The aircraft thrust (in N)
    mass : float
        The mass of aircraft (in kg)
    aircraft_state : str
        The state of the aircraft

    Returns
    -------
    float:
        The rate of change of mass (in kg/s)
    """
    mass_in_tonnes = mass / 1000.0
    if mass_in_tonnes < bada_parameters["m_max"] and mass_in_tonnes > bada_parameters["m_min"]:
        tas_in_knots = true_air_speed * convert.MPS_TO_KT
        Hp_in_feet = pressure_height * convert.M_TO_FT
        thrust_in_kilonewtons = thrust / 1000

        if bada_parameters["engtype"] == "Jet" or bada_parameters["engtype"] == "Turboprop":
            if bada_parameters["engtype"] == "Jet":
                eta = bada_parameters["Cf1"] * (1.0 + (tas_in_knots / bada_parameters["Cf2"]))
            elif bada_parameters["engtype"] == "Turboprop":
                eta = bada_parameters["Cf1"] * (1.0 - (tas_in_knots / bada_parameters["Cf2"])) * (tas_in_knots / 1000.0)
            f_min = bada_parameters["Cf3"] * (1.0 - Hp_in_feet / bada_parameters["Cf4"])
            f_nom = eta * thrust_in_kilonewtons
        elif bada_parameters["engtype"] == "Piston":
            f_min = bada_parameters["Cf3"]
            f_nom = bada_parameters["Cf1"]
        else:
            raise Exception("BADA engine type must be one of ['Jet', 'Turboprop', 'Piston']")

        if aircraft_state == "cruise":
            fuel_flow_rate = max(f_nom * bada_parameters["Cf_cruise"], f_min)
        else:
            fuel_flow_rate = max(f_nom, f_min)

        romc = -1.0 * fuel_flow_rate / 60  # Convert from kg/min to kg/s

    else:
        romc = 0.0

    return romc




[docs]def reduced_climb_power_correction(bada_parameters: dict, pressure_height: float, mass: float) -> float:
    """
    Calculates the reduced power correction to the ROCD equation. The relevant
    equations are 3.8-1 to 3.8-2 in EEC Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    bada_parameters : dict
        Dictionary or aircraft parameters from the BADA database.
    pressure_height : float
        The pressure height (in m)
    mass : float
        The actual mass of the aircraft (in kg)

    Returns
    -------
    float:
        The reduced power correction factor.
    """
    if pressure_height < 0.8 * bada_parameters["h_max"] * convert.FT_TO_M:
        if bada_parameters["engtype"] == "Jet":
            c_red = bada_parameters["Cred_jet"]
        elif bada_parameters["engtype"] == "Piston":
            c_red = bada_parameters["Cred_piston"]
        elif bada_parameters["engtype"] == "Turboprop":
            c_red = bada_parameters["Cred_turboprop"]
        else:
            raise Exception("BADA engine type must be one of ['Jet', 'Turboprop', 'Piston']")
        c_pow_red = 1.0 - c_red * (bada_parameters["m_max"] - (mass / 1000)) / (
            bada_parameters["m_max"] - bada_parameters["m_min"]
        )
    else:
        c_pow_red = 1.0
    return c_pow_red




[docs]def energy_share_factor(flight_condition: str, mach: float, temperature: float, delta_t: float) -> float:
    """
    Given a mach number calculate energy share factor. See equations
    3.2-8 to 3.2-11 in EEC Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    The flight_condition : str
        The flight condition of the aircraft
    mach : float
        The aircrafts speed as a mach number (dimensionless)
    temperature: float
        The atmospheric temperature at height of the aircraft (in K)
    delta_t : float
        The BADA atmosphere temperature differential at a pressure
        altitude of zero (in K)

    Returns
    -------
    float
        The energy share factor in BADA ROCD equation (dimensionless)

    Raises
    ------
    ValueError
        If the aircraft state is not recognised
    """

    if flight_condition == "constant mach above tropopause":
        f_m = 1.0

    elif flight_condition == "constant mach below tropopause":
        f_a = (
            1
            + 0.5
            * (constants.KAPPA * constants.R_GAS * constants.BETA_T / constants.G_ACC)
            * (1.0 - delta_t / temperature)
            * mach**2
        )
        f_m = 1.0 / f_a

    elif flight_condition == "constant cas above tropopause":
        f_b = 1.0 + 0.5 * (constants.KAPPA - 1.0) * mach**2
        f_m = 1.0 / (1.0 + f_b - f_b ** (-1.0 / (constants.KAPPA - 1.0)))

    elif flight_condition == "constant cas below tropopause":
        f_a = (
            1
            + 0.5
            * (constants.KAPPA * constants.R_GAS * constants.BETA_T / constants.G_ACC)
            * (1.0 - delta_t / temperature)
            * mach**2
        )
        f_b = 1.0 + 0.5 * (constants.KAPPA - 1.0) * mach**2
        f_m = 1.0 / (f_a + f_b - f_b ** (-1.0 / (constants.KAPPA - 1.0)))

    elif flight_condition == "acceleration in climb":
        f_m = 0.3

    elif flight_condition == "deceleration in climb":
        f_m = 1.7

    elif flight_condition == "acceleration in descent":
        f_m = 1.7

    elif flight_condition == "deceleration in descent":
        f_m = 0.3

    else:
        raise ValueError(
            "BADA flight condition must be one of ['constant mach above tropopause', \
            'constant mach below tropopause', 'constant cas above tropopause', \
            'constant cas below tropopause', 'acceleration in climb', 'acceleration in descent', \
            'deceleration in climb', 'deceleration in deescent' "
        )
    return f_m




[docs]def get_thrust(
    bada_parameters: dict, pressure_height: float, true_air_speed: float, delta_t: float, aircraft_state: str
) -> float:
    """
    This function computes a thrust value depending on the aircraft state. Here
    we use equations 3.7-1 to 3.7-12 in EEC Technical/Scientific Report No.
    2010-003.

    For:
    - climbs return maximum climb thrust
    - cruise phase return maximum cruise thrust
    - descent, approach and landing use expressions 3.7-9 to 3.7-12

    Parameters
    ----------
    bada_parameters : dict
        Dictionary or aircraft parameters from the BADA database.
    pressure_height : float
        The pressure height (in m)
    true_air_speed : float
        The true air speed (in m/s)
    delta_t : float
        The BADA atmosphere temperature differential at a pressure
        altitude of zero (in K)
    aircraft_state : str
        The state of the aircraft

    Returns
    -------
    float
        The maximum climb thrust (in N)
    """

    pressure_height = pressure_height * convert.M_TO_FT
    true_air_speed = true_air_speed * convert.MPS_TO_KT

    if bada_parameters["engtype"] == "Jet":
        T_climb_max_isa = bada_parameters["CTC"][0] * (
            1.0 - (pressure_height / bada_parameters["CTC"][1]) + bada_parameters["CTC"][2] * pressure_height**2
        )

    elif bada_parameters["engtype"] == "Turboprop":
        T_climb_max_isa = (
            bada_parameters["CTC"][0] * (1 - pressure_height / bada_parameters["CTC"][1]) / true_air_speed
            + bada_parameters["CTC"][2]
        )

    elif bada_parameters["engtype"] == "Piston":
        T_climb_max_isa = (
            bada_parameters["CTC"][0] * (1 - pressure_height / bada_parameters["CTC"][1])
            + bada_parameters["CTC"][2] / true_air_speed
        )

    else:
        raise Exception("BADA engine type must be one of ['Jet', 'Turboprop', 'Piston']")

    delta_t_eff = delta_t - bada_parameters["CTC"][3]
    if delta_t_eff * bada_parameters["CTC"][4] < 0:
        non_isa_thrust_correction = 1
    elif delta_t_eff * bada_parameters["CTC"][4] > 0.4:
        non_isa_thrust_correction = 1 - 0.4
    else:
        non_isa_thrust_correction = 1 - delta_t_eff * bada_parameters["CTC"][4]

    maximum_climb_thrust = T_climb_max_isa * non_isa_thrust_correction
    if aircraft_state == "climb" or aircraft_state == "take_off":
        thrust = maximum_climb_thrust
    elif aircraft_state == "cruise":
        # Set this equal to maximum cruise thrust
        thrust = 0.95 * maximum_climb_thrust
    elif aircraft_state == "descent":
        if pressure_height > bada_parameters["Hp_des"]:
            thrust = bada_parameters["CTdes_high"] * maximum_climb_thrust
        else:
            thrust = bada_parameters["CTdes_low"] * maximum_climb_thrust
    elif aircraft_state == "approach":
        thrust = bada_parameters["CTdes_app"] * maximum_climb_thrust
    elif aircraft_state == "landing":
        thrust = bada_parameters["CTdes_land"] * maximum_climb_thrust
    else:
        raise Exception("BADA aircraft state must be one of ['climb', 'cruise', 'descent', 'approach', 'landing']")

    return thrust

[docs]def get_drag(
    bada_parameters: dict, mass: float, true_air_speed: float, density: float, banking_angle: float, aircraft_state: str
) -> float:
    """
    This function computes the drag on an aircraft. Here we use equations 3.6-1
    to 3.6-5 in EEC Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    bada_parameters : dict
        Dictionary or aircraft parameters from the BADA database.
    mass : float
        The mass of the aircraft (in kg)
    true_air_speed : float
        The true air speed (in m/s)
    density : float
        The density of the air.
    banking angle : float
        The banking angle of the airplane.
    aircraft_state : str
        The state of the aircraft

    Returns
    -------
    float
        The drag on the airplane (in N)
    """
    C_L = 2 * mass * constants.G_ACC / (density * (true_air_speed**2) * bada_parameters["S"] * np.cos(banking_angle))
    if aircraft_state == "climb" or aircraft_state == "cruise" or aircraft_state == "descent":
        C_D = bada_parameters["CD0_cr"] + bada_parameters["CD2_cr"] * C_L**2
    elif aircraft_state == "approach":
        C_D = bada_parameters["CD0_ap"] + bada_parameters["CD2_ap"] * C_L**2
    elif aircraft_state == "landing":
        C_D = bada_parameters["CD0_ld"] + bada_parameters["CD0_gear"] + bada_parameters["CD2_ld"] * C_L**2
    else:
        raise Exception("BADA aircraft state must be one of ['climb', 'cruise', 'descent', 'approach', 'landing']")
    return C_D * density * (true_air_speed**2) * bada_parameters["S"] / 2.0
