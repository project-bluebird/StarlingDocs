import numpy as np
import typing
from scipy.integrate import solve_ivp

import starling.core as core
from . import constants
from . import convert

DEFAULT_MASS = 4.0e5
DEFAULT_THRUST = 1.1e6
DEFAULT_DRAG = 1.3e6



[docs]def change_in_pressure_height(
    aircraft: Aircraft, atmos: Atmosphere, aircraft_state: str, time_interval: float
) -> float:
    """
    For a given aircraft find its change in pressure height over a given
    time interval using a fourth order Runge-Kutta integration. No automatic
    step size control is implemented. The integration time step is for now
    set to be 1% of the provided time interval.

    Parameters
    ----------
    aircraft : Aircraft
        The aircraft for which to find the change in pressure height.
    atmos : Atmosphere
        The BADA atmosphere used to represent the atmosphere at lat and lon of
        the aircraft.
    aircraft_state : str
        The state of the aircraft.
    time_interval : float
        The time interval in seconds

    Returns
    -------
    float
        The change in pressure height (in meters)
    """

    pressure_height_0 = aircraft.fl * convert.FL_TO_M

    solution = solve_ivp(
        lambda t, y: dHp_dt(t, y, atmos, aircraft, aircraft_state),
        (0, time_interval),
        np.array([pressure_height_0]),
    )

    return solution.y[0, -1] - pressure_height_0




[docs]def dHp_dt(
    time: float,
    pressure_height: float,
    atmos: Atmosphere,
    aircraft: Aircraft,
    aircraft_state: str,
) -> float:
    """
    For a given aircraft find its rate of climb or descent (ROCD i.e. dHp/dt).
    Here we use equation 3.2-7 in ECC Technical/Scientific Report No. 2010-003

    Parameters
    ----------
    time : float
        Dummy time variable for solve_ivp algorithm
    pressure_height : float
        The pressure_height level of the aircraft
    atmos : Atmosphere
        The BADA atmosphere used to represent the atmosphere at lat and lon of
        the aircraft.
    aircraft : Aircraft or BadaAircraft
        An Aircraft or BadaAircraft instance.
    aircraft_state : str
        The state of the aircraft.

    Returns
    -------
    float
        The rate of climb or descent (in m/s)
    """

    if isinstance(aircraft, core.BadaAircraft):
        bada_parameters = aircraft.bada_parameters
    else:
        bada_parameters = None

    temperature = atmos.temperature_from_fl(pressure_height * convert.M_TO_FL)
    density = atmos.density_from_fl(pressure_height * convert.M_TO_FL)
    delta_t = atmos.delta_T

    aircraft_speed_in_m_per_sec = aircraft.speed * convert.KT_TO_MPS
    true_air_speed = gs_to_tas(aircraft_speed_in_m_per_sec)

    mach = tas_to_mach(true_air_speed, temperature)

    mass = bada_parameters["m_ref"] * 1000 if bada_parameters is not None else DEFAULT_MASS

    thrust, drag = get_thrust_and_drag(bada_parameters, pressure_height, true_air_speed, delta_t, density)

    rocd = (
        (1.0 - delta_t / temperature)
        * ((thrust - drag) * true_air_speed / (mass * constants.G_ACC))
        * energy_share_factor(aircraft_state, mach, temperature, delta_t)
    )

    return rocd




[docs]def gs_to_tas(ground_speed: float) -> float:
    """
    Convert ground speed (GS) to true air speed (TAS).
    This is merely a holding routine. For the moment the airspeed and
    ground speed are set to be equal. I.e. we are ignoring any
    correction due to windspeed at the location of the aircraft.

    Parameters
    ----------
    ground_speed : float
        The ground speed of the aircraft (in m/s)

    Returns
    -------
    float
        The true air speed of the aircraft (in m/s)
    """
    true_air_speed = ground_speed
    return true_air_speed




[docs]def tas_to_mach(true_air_speed: float, temperature: float) -> float:
    """
    Converts the true airspeed (TAS) to a mach number.
    See equation 3.1-26 in ECC Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    true_air_speed : float
        The true air speed of the aircraft (in m/s)
    temperature : float
        The atmospheric temperature at height of the aircraft (in K)

    Returns
    -------
    float
        The aircrafts speed as a mach number (dimensionless)
    """
    mach = true_air_speed / np.sqrt(constants.KAPPA * constants.R_GAS * temperature)
    return mach




[docs]def energy_share_factor(aircraft_state: str, mach: float, temperature: float, delta_t: float) -> float:
    """
    Given a mach number calculate energy share factor. See equations
    3.2-8 to 3.2-11 in ECC Technical/Scientific Report No. 2010-003.

    Parameters
    ----------
    aircraft_state : str
        The state of the aircraft
    mach : float
        The aircrafts speed as a mach number (dimensionless)
    temperature: float
        The atmospheric temperature at height of the aircraft (in K)
    delta_t : float
        The BADA atmosphere temperature differential at a pressure
        altitude of zero (in K)

    Returns
    -------
    float
        The energy share factor in BADA ROCD equation (dimensionless)

    Raises
    ------
    ValueError
        If the aircraft state is not recognised
    """
    if aircraft_state == "constant mach above tropopause":
        f_m = 1.0

    elif aircraft_state == "constant mach below tropopause":
        f_a = (
            1
            + 0.5
            * (constants.KAPPA * constants.R_GAS * constants.BETA_T / constants.G_ACC)
            * (1.0 - delta_t / temperature)
            * mach ** 2
        )
        f_m = 1.0 / f_a

    elif aircraft_state == "constant cas above tropopause":
        f_b = 1.0 + 0.5 * (constants.KAPPA - 1.0) * mach ** 2
        f_m = 1.0 / (1.0 + f_b - f_b ** (-1.0 / (constants.KAPPA - 1.0)))

    elif aircraft_state == "constant cas below tropopause":
        f_a = (
            1
            + 0.5
            * (constants.KAPPA * constants.R_GAS * constants.BETA_T / constants.G_ACC)
            * (1.0 - delta_t / temperature)
            * mach ** 2
        )
        f_b = 1.0 + 0.5 * (constants.KAPPA - 1.0) * mach ** 2
        f_m = 1.0 / (f_a + f_b - f_b ** (-1.0 / (constants.KAPPA - 1.0)))

    elif aircraft_state == "acceleration in climb":
        f_m = 0.3

    elif aircraft_state == "deceleration in climb":
        f_m = 1.7

    elif aircraft_state == "acceleration in descent":
        f_m = 1.7

    elif aircraft_state == "deceleration in descent":
        f_m = 0.3

    else:
        raise ValueError("Unrecognised case in BADA model energy share factor")

    return f_m




[docs]def get_thrust_and_drag(
    bada_parameters: typing.Union[dict, None],
    pressure_height: float,
    true_air_speed: float,
    delta_t: float,
    density: float,
) -> typing.Tuple[float, float]:
    """
    Returns the aircraft's physical parameters: mass, thrust and drag given the
    aircraft type. Purely a dummy routine for now.

    Parameters
    ----------
    bada_parameters : dict or None
        The aircraft parameters from the BADA database.
    pressure_height : float
        The pressure_height level of the aircraft
    true_air_speed : float
        The true air speed of the aircraft (in m/s)
    delta_t : float
        The BADA atmosphere temperature differential at a pressure
        altitude of zero (in K)
    density : float
        The density of the air

    Returns
    -------
    Tuple
        Tuple of two floats containing in order, the thrust (in N) and the drag (in N)
    """

    if bada_parameters is None:
        thrust = DEFAULT_THRUST
        drag = DEFAULT_DRAG
    else:
        maximum_climb_thrust = get_maximum_climb_thrust(bada_parameters, pressure_height, true_air_speed, delta_t)
        thrust = get_thrust(maximum_climb_thrust)

        banking_angle = get_banking_angle()
        drag = get_drag(bada_parameters, true_air_speed, density, banking_angle)
    return thrust, drag




[docs]def get_maximum_climb_thrust(bada_parameters: dict, pressure_height: float, true_air_speed: float, delta_t: float):
    """
    This function computes the maximum climb thrust as described in 
    the BADA manual.
    
    Parameters
    ----------
    bada_parameters : dict
        Dictionary or aircraft parameters from the BADA database.
    pressure_height : float
        The pressure height (in m)
    true_air_speed : float
        The true air speed (in m/s)
    delta_t : float
        The BADA atmosphere temperature differential at a pressure
        altitude of zero (in K)

    Returns
    -------
    float
        The maximum climb thrust (in N)
    """

    pressure_height = pressure_height * convert.M_TO_FT
    true_air_speed = true_air_speed * convert.MPS_TO_KT
    
    if bada_parameters["engtype"] == "Jet":
        T_climb_max_isa = bada_parameters["CTC"][0] * (
            1 - pressure_height / bada_parameters["CTC"][1] + bada_parameters["CTC"][2] * pressure_height ** 2
        )

    elif bada_parameters["engtype"] == "Turboprop":
        T_climb_max_isa = (
            bada_parameters["CTC"][0] * (1 - pressure_height / bada_parameters["CTC"][1]) / true_air_speed
            + bada_parameters["CTC"][2]
        )

    elif bada_parameters["engtype"] == "Piston":
        T_climb_max_isa = (
            bada_parameters["CTC"][0] * (1 - pressure_height / bada_parameters["CTC"][1])
            + bada_parameters["CTC"][2] / true_air_speed
        )

    delta_t_eff = delta_t - bada_parameters["CTC"][3]

    if delta_t_eff * bada_parameters["CTC"][4] < 0:
        non_isa_thrust_correction = 1
    elif delta_t_eff * bada_parameters["CTC"][4] > 0.4:
        non_isa_thrust_correction = 1 - 0.4
    else:
        non_isa_thrust_correction = 1 - delta_t_eff * bada_parameters["CTC"][4]

    return T_climb_max_isa * non_isa_thrust_correction

[docs]def get_drag(bada_parameters: dict, true_air_speed: float, density: float, banking_angle: float) -> float:
    """
    This function computes the drag on an aircraft as described in the BADA manual.
    
    Parameters
    ----------
    bada_parameters : dict
        Dictionary or aircraft parameters from the BADA database.
    true_air_speed : float
        The true air speed (in m/s)
    density : float
        The density of the air.
    banking angle : float
        The abnking angle of the airplane.

    Returns
    -------
    float
        The drag on the airplane (in N)
    """

    C_L = (
        2
        * bada_parameters["m_ref"]
        * 1000
        * constants.G_ACC
        / (density * true_air_speed ** 2 * bada_parameters["S"] * np.cos(banking_angle))
    )
    C_D = bada_parameters["CD0_cr"] + bada_parameters["CD2_cr"] * C_L ** 2

    return C_D * density * true_air_speed ** 2 * bada_parameters["S"] / 2
