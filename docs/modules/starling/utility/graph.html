from collections import defaultdict
import typing



[docs]def build_from_edges(edges: typing.List[typing.Tuple[str, str]]) -> typing.Dict[str, str]:
    """
    Build an undirected graph from a list of edges.
    """

    graph = defaultdict(list)
    for (a, b) in edges:
        graph[a].append(b)
        graph[b].append(a)

    for node in graph:
        graph[node] = list(dict.fromkeys(graph[node]))

    return graph




[docs]def shortest_path(graph: typing.Dict[str, typing.List[str]], start: str, end: str, path: typing.List[str] = []):
    """
    Find the shortest path between two nodes.
    """

    path = path + [start]

    if start == end:
        return path

    shortest = None
    for node in graph[start]:
        if node not in path:
            new_path = shortest_path(graph, node, end, path)
            if new_path:
                if (not shortest) or (len(new_path) < len(shortest)):
                    shortest = new_path

    return shortest
