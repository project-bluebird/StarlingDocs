    """

[docs]def intersection(segment0: tuple[Pos2D, Pos2D], segment1: tuple[Pos2D, Pos2D]) -> Pos2D:
    """
    Determine the intersection point of two line segments defined by ab and cd.
    """

    from ..core import Pos2D

    (a, b) = segment0
    (c, d) = segment1

    lat = (
        (((b.lat * a.lon) - (a.lat * b.lon)) * (d.lat - c.lat))
        - (((d.lat * c.lon) - (c.lat * d.lon)) * (b.lat - a.lat))
    ) / (((b.lat - a.lat) * (d.lon - c.lon)) - ((d.lat - c.lat) * (b.lon - a.lon)))
    lon = (
        (((b.lat * a.lon) - (a.lat * b.lon)) * (d.lon - c.lon))
        + (((d.lat * c.lon) - (c.lat * d.lon)) * (b.lon - a.lon))
    ) / (((b.lat - a.lat) * (d.lon - c.lon)) - ((d.lat - c.lat) * (b.lon - a.lon)))

    return Pos2D(lat, lon)




[docs]def rotate(pos: Pos2D, theta: float) -> Pos2D:
    """
    Rotate the given position around the origin by the given angle [rad].
    """

    from ..core import Pos2D

    theta *= -math.pi / 180.0

    lat = (pos.lat * math.cos(theta)) - (pos.lon * math.sin(theta))
    lon = (pos.lat * math.sin(theta)) + (pos.lon * math.cos(theta))

    return Pos2D(lat, lon)




[docs]def rotate_about(pos: Pos2D, theta: float, pivot: Pos2D) -> Pos2D:
    """
    Rotate the given position around a pivot by the given angle [rad].
    """

    from ..core import Pos2D

    theta *= -math.pi / 180.0

    lat = (((pos.lat - pivot.lat) * math.cos(theta)) - ((pos.lon - pivot.lon) * math.sin(theta))) + pivot.lat
    lon = (((pos.lat - pivot.lat) * math.sin(theta)) + ((pos.lon - pivot.lon) * math.cos(theta))) + pivot.lon

    return Pos2D(lat, lon)




[docs]def min_distance_between_segments(segment0: tuple[Pos2D, Pos2D], segment1: tuple[Pos2D, Pos2D]) -> float:
    """
    Determine the minium distance between two segments.
    """

    (a, b) = segment0
    (c, d) = segment1

    a = np.array(a)
    b = np.array(b)
    c = np.array(c)
    d = np.array(d)

    mag0 = np.linalg.norm(b - a)
    mag1 = np.linalg.norm(d - c)

    inv0 = np.zeros(3)
    if mag0 > 0.0:
        inv0 = (b - a) / mag0
    inv1 = np.zeros(3)
    if mag1 > 0.0:
        inv1 = (d - c) / mag1

    cross = np.cross(inv0, inv1)
    denom = np.linalg.norm(cross) ** 2

    if not denom:
        p = np.dot(inv0, (c - a))
        q = np.dot(inv0, (d - a))

        if p <= 0.0 >= q:
            if np.absolute(p) < np.absolute(q):
                return np.linalg.norm(a - c) * convert.DEG_TO_NMI
            return np.linalg.norm(a - d) * convert.DEG_TO_NMI

        elif p >= mag0 <= q:
            if np.absolute(p) < np.absolute(q):
                return np.linalg.norm(b - c) * convert.DEG_TO_NMI
            return np.linalg.norm(b - d) * convert.DEG_TO_NMI

        return np.linalg.norm(((p * inv0) + a) - c) * convert.DEG_TO_NMI

    t = c - a
    det_0 = np.linalg.det([t, inv1, cross])
    det_1 = np.linalg.det([t, inv0, cross])

    t0 = det_0 / denom
    t1 = det_1 / denom

    s0 = a + (inv0 * t0)
    s1 = c + (inv1 * t1)

    if t0 < 0.0:
        s0 = a
    elif t0 > mag0:
        s0 = b

    if t1 < 0.0:
        s1 = c
    elif t1 > mag1:
        s1 = d

    if (t0 < 0.0) or (t0 > mag0):
        dot = np.dot(inv1, (s0 - c))
        if dot < 0.0:
            dot = 0.0
        elif dot > mag1:
            dot = mag1
        s1 = c + (inv1 * dot)

    if (t1 < 0.0) or (t1 > mag1):
        dot = np.dot(inv0, (s1 - a))
        if dot < 0.0:
            dot = 0.0
        elif dot > mag0:
            dot = mag0
        s0 = a + (inv0 * dot)

    return np.linalg.norm(s0 - s1) * convert.DEG_TO_NMI




[docs]def compute_distance_matrix(x: np.ndarray, method: str = "numpy"):
    """Compute distance matrix using numpy or JAX

    Parameters
    ----------
    x : np.ndarray
        rows: samples/points
        columns: coordinates
        e.g., x = np.random.randn(250, 3) in which we have
            250 points defined by 3 coordinates
    method : str, optional
        numpy or jax, by default "numpy"
    """
    if method.lower() in ["numpy", "np"]:
        return compute_distance_np(x)
    elif method.lower() in ["jax"]:
        raise NotImplemented(f"The method: {method} is not implemented!")
    else:
        raise NotImplemented(f"The method: {method} is not implemented!")




[docs]def compute_distance_np(x: np.ndarray) -> np.ndarray:
    """Compute distance matrix using NumPy

    Parameters
    ----------
    x : np.ndarray
        rows: samples/points
        columns: coordinates
        e.g., x = np.random.randn(250, 3) in which we have
            250 points defined by 3 coordinates
    """
    dist_mat = np.sqrt(np.sum((x[:, None, :] - x[None, :, :]) ** 2, axis=-1))
    return dist_mat
