import typing
import random
import numpy as np

from collections import defaultdict

from .aircraft_generator import AircraftGenerator
from ..core import Aircraft, Airspace, Contract, FlightPlan, Route, Pos2D



[docs]class Tactical(AircraftGenerator):
    """
    Aircraft generator for simple tactical scenarios:
    - configurable number of Aircraft and balance of climbers/descenders/overfliers
    - randomly selected entry and exit Contracts
    - randomly generated speeds
    - ensures that no two Aircraft have the same entry contract (same Fix, Flight Level and time)
    """


[docs]    def __init__(
        self,
        num_aircraft: int,
        balance: typing.List[float] = [1 / 3, 1 / 3, 1 / 3],
        speed_range: typing.List[float] = [400.0, 400.0],
        time_entry_gap: float = 5,
    ):
        """
        Construct a new instance.

        Parameters
        ----------
        num_aircraft: float
            Number of Aircraft to generate.
        balance: list[float, float, float]
            Probabilities of any given Aircraft being one of climber/descender/overflier.
            The probabilities have to sum to 1 (Multinomial distribution parameter).
        speed_range: list[float, float]
            Optional range of [min,max] speeds from which to randomly generate Aircraft speed.
            If not provided, speed of all Aircraft is set to 400.
        time_entry_gap: float
            Optional amount of time in seconds that must be maintained between two Aircraft entry Contracts if they are at the same Fix and FL.
        """

        if num_aircraft <= 0:
            raise ValueError("Number of Aircraft must be positive.")

        if len(balance) != 3:
            raise ValueError("The balance parameter must be a list of length 3.")

        if sum(balance) != 1:
            raise ValueError("Balance probabilities must sum to 1.")

        if len(speed_range) != 2:
            raise ValueError("Speed range must be a list of length 2.")

        if time_entry_gap < 0:
            raise ValueError("Time entry gap cannot be negative.")

        self.num_aircraft = num_aircraft
        self.balance = balance
        self.speed_range = speed_range
        self.time_entry_gap = time_entry_gap



[docs]    def check_entry_contract(
        self, new_aircraft: Aircraft, entries: typing.Dict[Pos2D, list], airspace: Airspace
    ) -> Aircraft:
        """
        Check if proposed new Aircraft has the same entry Contract as any of the existing Aircraft (same entry Fix, FL and time).
        If yes, increase entry time by self.time_entry_gap.

        Parameters
        ----------
        aircraft: Aircraft
            A new Aircraft with a proposed entry Contract.
        entries: dict[Pos2D, Contract]
            A dictionary of the already agreed entry Contracts by Fix in the format {fix_pos: [Contract]}.
        airspace: Airspace
            The Airspace the Aircraft is flying through.

        Returns
        ----------
        Aircraft
            An Aircraft whose entry Contract does not clash with any of the already agreed entry Contracts.
        """

        entry_fix = airspace.route_entry_fix(new_aircraft.flight_plan.route)

        # do nothing if this is the first aircraft
        if len(entries) == 0:
            return new_aircraft

        # do nothing if no other Aircraft is entering at this Fix
        if entry_fix not in list(entries.keys()):
            return new_aircraft

        # do nothing if no other Aircraft has the same time and FL contract at this Fix
        contracts_at_fix = entries[entry_fix]
        entry_times = [
            contract.time for contract in contracts_at_fix if contract.fl == new_aircraft.flight_plan.entry.fl
        ]
        if new_aircraft.flight_plan.entry.time not in entry_times:
            return new_aircraft

        # otherwise, get the last Aircraft at Fix and FL and increase the entry time to be after
        new_entry_time = max(entry_times) + self.time_entry_gap
        time_increase = new_entry_time - new_aircraft.flight_plan.entry.time
        new_aircraft.flight_plan.entry.time = new_entry_time
        new_aircraft.flight_plan.exit.time += time_increase
        new_aircraft.start_time += time_increase

        return new_aircraft



[docs]    def generate_aircraft(self, airspace: Airspace, routes: typing.List[Route]) -> typing.Dict[str, Aircraft]:
        """
        Generate Aircraft for the given Airspace.

        Parameters
        ----------
        airspace: Airspace
            The Airspace the Aircraft are flying through.
        routes: list[Route]
            The available Routes in the Airspace (choose one at random for each Aircraft Route).

        Returns
        ----------
        Dict[str, Aircraft]
            A dictionary with (callsign, Aircraft) items.
        """

        # split aircraft into climbers, descenders and overfliers
        climbers, descenders, overfliers = np.random.multinomial(self.num_aircraft, self.balance)
        journey_type = ["climb"] * climbers + ["descend"] * descenders + ["overfly"] * overfliers
        random.shuffle(journey_type)

        aircraft = {}
        # keep track of entry contracts to avoid clashes
        entries = defaultdict(list)

        for i in range(self.num_aircraft):

            route = random.choice(routes)
            entry_FLs = airspace.get_FLs_entry(route)
            exit_FLs = airspace.get_FLs_exit(route)

            journey = journey_type[i]
            if journey == "overfly":
                # FLs that are valid at both exit and entry for overfliers
                constant_FLs = list(set.intersection(set(entry_FLs), set(exit_FLs)))
                fl1 = fl2 = random.choice(constant_FLs)
            elif journey == "climb":
                # fl1 < fl2
                # choose start FL so that Aircraft has somewhere to climb to
                fl1 = random.choice(np.arange(entry_FLs[0], min(entry_FLs[-1], exit_FLs[-1]), 10))
                # choose exit to be higher than fl1 AND within exit FL limits
                fl2 = random.choice(np.arange(max(fl1 + 10, exit_FLs[0]), exit_FLs[-1] + 10, 10))
            elif journey == "descend":
                # fl1 > fl2
                # choose start FL so that Aircraft has somewhere to descend to
                fl1 = random.choice(np.arange(max(entry_FLs[0], exit_FLs[1]), entry_FLs[-1] + 10, 10))
                # choose exit to be lower than fl1 AND within exit FL limits
                fl2 = random.choice(np.arange(exit_FLs[0], min(fl1, exit_FLs[-1] + 10), 10))

            callsign = f"AIR{len(aircraft)}"
            speed = speed = self.speed_range[0] + (self.speed_range[1] - self.speed_range[0]) * np.random.uniform()
            vertical_speed = 0.0
            fix1 = airspace.fixes.places[route.names[0]]
            heading = fix1.bearing_to(airspace.fixes.places[route.names[1]])
            pos = fix1.pos3d(fl1)

            start_t = 0.0
            entry_time, exit_time = self.get_contract_times(speed, route, airspace, start_t)
            entry = Contract(fl1, entry_time)
            exit = Contract(fl2, exit_time)
            flight_plan = FlightPlan(entry, exit, route)

            # check entry contract and make sure it doesn't clash with another Aircraft entry contract
            proposed_aircraft = Aircraft(
                flight_plan, pos.lat, pos.lon, pos.fl, heading, speed, vertical_speed, start_time=start_t
            )
            checked_aircraft = self.check_entry_contract(proposed_aircraft, entries, airspace)
            aircraft[callsign] = checked_aircraft

            # save entry contract
            entry_fix = airspace.route_entry_fix(checked_aircraft.flight_plan.route)
            entries[entry_fix].append(checked_aircraft.flight_plan.entry)

        return aircraft
