[docs]    def get_contract_times(
        self, speed: float, route: Route, airspace: Airspace, start_time: float = 0.0
    ) -> typing.Tuple[float, float]:
        """
        Helper function to generate [entry, exit] Contract times for Aircraft (based on Route length and speed).

        Parameters
        ----------
        speed: float
            The speed the Aircraft is travelling at.
        route: Route
            The Route the Aircraft is to fly.
        airspace: Airspace
            The Airspace the Aircraft is flying through.
        start_time: float
            The time at which the Aircraft starts travelling on the Route.

        Returns
        ----------
        list[float]
            Time of the [entry, exit] Contract of the Aircraft.
        """

        start_pos = airspace.fixes.places[route.names[0]]

        entry_fix = airspace.route_entry_fix(route)
        entry_fix_pos = list(entry_fix.values())[0]

        exit_fix = airspace.route_exit_fix(route)
        exit_fix_pos = list(exit_fix.values())[0]

        dist_to_entry = start_pos.distance(entry_fix_pos)
        dist_to_exit = dist_to_entry + entry_fix_pos.distance(exit_fix_pos)

        entry_time = dist_to_entry / speed * 3600 + start_time
        exit_time = dist_to_exit / speed * 3600 + start_time

        return entry_time, exit_time
