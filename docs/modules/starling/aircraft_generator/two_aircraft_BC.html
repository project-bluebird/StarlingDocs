[docs]    def level_gen(self, min_fl=60, max_fl=400):
        """generate random flight level"""
        level = random.randrange(min_fl, max_fl, 10)
        return level


    def gen_random_contracts(self, min_fl=60, max_fl=400):
        in_level = self.level_gen()

        # 1/3 of time make in_level = out_level to create overflier scenario
        if random.randint(1, 3) == 1:
            out_level = in_level
        else:
            out_level = self.level_gen()
        return [Contract(in_level, 0.0), Contract(out_level, self.total_time)]


[docs]    def generate_aircraft_events(self, airspace: Airspace, routes: typing.List[Route]) -> typing.List[AddAircraftEvent]:
        """
        Generate two Aircraft for the given Airspace.
        """

        aircraft_events = []

        route_fwd = random.choice(routes)
        route_rev = Route(route_fwd.names[::-1])
        routes = [route_fwd, route_rev]

        contracts_fwd = self.gen_random_contracts()
        contracts_rev = self.gen_random_contracts()

        contracts = [contracts_fwd, contracts_rev]

        # speed is in knots i.e., nmi per hour
        speed = 400
        vertical_speed = 0

        for i in range(2):

            callsign = f"AIR{i+1}"

            route = routes[i]
            entry, exit = contracts[i]
            flight_plan = FlightPlan(entry, exit, route)

            fix1 = airspace.fixes.places[route.names[0]]
            heading = fix1.bearing_to(airspace.fixes.places[route.names[1]])

            pos = fix1.pos3d(entry.fl)

            aircraft = Aircraft(flight_plan, pos.lat, pos.lon, pos.fl, heading, speed, vertical_speed)
            aircraft_events.append(AddAircraftEvent(callsign, 0.0, aircraft))

        return aircraft_events
