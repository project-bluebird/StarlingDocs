[docs]    def __init__(self, total_time: float, num_aircraft: int):
        """
        Construct a new instance.

        Parameters
        ----------
        total_time: float
            Time period [sec] within which all Aircraft enter the Airspace.
            The Aircraft entry times are quasi-regularly spaced out within this interval.
        num_aircraft: float
            Number of Aircraft to generate.
        """

        if total_time <= 0.0:
            raise ValueError("Total time must be positive.")

        if num_aircraft <= 0:
            raise ValueError("Number of Aircraft must be positive.")

        self.total_time = total_time
        self.num_aircraft = num_aircraft



[docs]    def generate_aircraft(self, airspace: Airspace, routes: typing.List[Route]) -> typing.Dict[str, Aircraft]:
        """
        Generate a set of Aircraft for the given Airspace.
        """

        entry_times = [random.random() for _ in range(self.num_aircraft)]
        total = sum(entry_times)
        t = -entry_times[0] * 0.5
        for i in range(len(entry_times)):
            t += entry_times[i]
            entry_times[i] = (t / total) * self.total_time

        aircraft = {}
        for entry_time in entry_times:
            flight_time = convert.time_str("00:30:00")
            route = random.choice(routes)

            # entry/exit flight level should be within the Airspace limits at those points
            entry_fl = random.choice(airspace.get_FLs_entry(route))
            exit_fl = random.choice(airspace.get_FLs_exit(route))
            entry = Contract(entry_fl, entry_time)
            exit = Contract(exit_fl, entry_time + flight_time)

            flight_plan = FlightPlan(entry, exit, route)
            pos = airspace.fixes.places[route.names[0]].pos3d(entry.fl)
            heading = airspace.fixes.places[route.names[0]].bearing_to(airspace.fixes.places[route.names[1]])
            speed = route.length(airspace.fixes) / ((exit.time - entry.time) / 3600.0)
            vertical_speed = 0

            aircraft[f"AIR{len(aircraft)}"] = Aircraft(
                flight_plan,
                pos.lat,
                pos.lon,
                pos.fl,
                heading,
                speed,
                vertical_speed,
            )

        return aircraft
