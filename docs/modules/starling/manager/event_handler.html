[docs]    def __init__(self, events: typing.Sequence[Event]):
        """
        Construct a new instance.

        Parameters
        ----------
        events: List[Event]
            List of Events to apply to the Environment.
        """

        # events to be carried out some time in the future. events are stored
        # internally in a dictionary to allow for O(1) access/removal
        self.upcoming: typing.Dict[int, Event] = {
            event_id: event for (event_id, event) in enumerate(events)
        }

        # events that are currently underway and should be carried out at each
        # time-step. For now this is unused but will be needed in the future.
        self.ongoing: typing.Dict[int, Event] = {}

        # events that have been completed
        self.completed: typing.Dict[int, Event] = {}



[docs]    @staticmethod
    def from_json(s: str) -> EventHandler:
        """
        Construct a new instance from a string in JSON format.

        Parameters
        ----------
        s: str
            A string representation of EventHandler in JSON/dict format.

        Returns
        ----------
        EventHandler
        """

        data = json.loads(s)

        upcoming: typing.Dict[int, Event] = {}
        ongoing: typing.Dict[int, Event] = {}
        completed: typing.Dict[int, Event] = {}

        for (key, storage) in zip(
            ["upcoming", "ongoing", "completed"],
            [upcoming, ongoing, completed],
        ):

            for event_id, event_data in data[key]:
                for event_type, kwargs in event_data.items():
                    # get the event's class
                    event_class = getattr(starling.events, event_type)

                    # instantiate it with the kwargs dictionary
                    storage[event_id] = event_class(**kwargs)

        # create an empty event handler and populate its events manually.
        # this allows for the preservation of the event ids
        eh = EventHandler([])
        eh.upcoming = upcoming
        eh.ongoing = ongoing
        eh.completed = completed

        return eh

[docs]    def forward(self, environment: Environment, time_step: float):
        """
        Evolve Environment for the given time step [sec] by applying Events.

        Parameters
        ----------
        environment: Environment
            The Environment to apply Events to.
        time_step: float
            Amount of time in seconds the Environment is evolving for.
        """

        episode_start = environment.time
        episode_end = episode_start + time_step

        # carry out ongoing (repeating) events
        for event_id, event in self.ongoing.items():
            environment = event.run(environment)

        # temporary dict to store which events to remove from the upcoming dict
        events_to_remove_from_upcoming = {}

        # carry out the new events
        for event_id, event in self.upcoming.items():
            if event.start_time >= episode_start and event.start_time <= episode_end:
                environment = event.run(environment)

                # move the event to its new status
                if event.repeated:
                    self.ongoing[event_id] = event
                else:
                    self.completed[event_id] = event

                # store the event_id to be removed
                events_to_remove_from_upcoming[event_id] = 1

        # finally, remove newly-started events from the upcoming event dict
        for event_id in events_to_remove_from_upcoming:
            del self.upcoming[event_id]

        return environment
