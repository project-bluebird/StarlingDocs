import typing

from collections import defaultdict

[docs]    def __init__(self, airspace: Airspace, event_handler: EventHandler, predictor: Predictor):
        """
        Construct a new instance.

        Parameters
        ----------
        airspace: Airspace
            The Airspace to be controlled.
        event_handler: EventHandler
            The EventHandler applies events to the internal environment of the EM.
        predictor: Predictor
            Aircraft Trajectory prediction used to evolve Aircraft.
        """

        self.environment = Environment(time=0.0, airspace=airspace, aircraft={})
        self.event_handler = event_handler
        self.predictor = predictor
        self.actions_to_apply = []
        # keep track of full trajectories travelled by all Aircraft
        self.control_points = defaultdict(list)



[docs]    def get_trajectory(self, callsign: str) -> Trajectory:
        """
        Return full Trajectory travelled by an Aircraft up to current Environment time.

        Parameters
        ----------
        callsign: str
            The Aircraft callsign.

        Returns
        ----------
        Trajectory
            The Aircraft Trajectory travelled so far.
        """

        if len(self.control_points[callsign]) >= 2:
            full_trajectory = np.concatenate(self.control_points[callsign])
            return Trajectory(full_trajectory)
        else:
            return None



[docs]    def evolve_aircraft(self, callsigns: typing.List[str], time_step: float) -> typing.Dict[str, Trajectory]:
        """
        Evolve the specified Aircraft in the Environment for time_step [sec].

        Parameters
        ----------
        callsigns: List[str]
            A list of Aircraft callsigns to evolve in time.
        time_step: float
            Amount of time in seconds to evolve the Environment for.

        Returns
        ----------
        Dict[str, Trajectory]
            The subtrajectories travelled during time_step. The dictionary keys are Aircraft callsigns.
        """

        subtrajectories = dict()

        for callsign in callsigns:
            subtrajectory = self.predictor.future_trajectory(
                callsign, self.environment, time_step, self.actions_to_apply
            )

            # if there is no trajectory, move on to the next Aircraft
            if subtrajectory is None:
                continue

            # save subtrajectory prediction and move the aircraft
            self.environment.aircraft[callsign].travel(subtrajectory)
            self.control_points[callsign].append(subtrajectory.locations)
            subtrajectories[callsign] = subtrajectory

            # keep track of what is requested in actions
            self.environment.aircraft[callsign].requested_heading = self.predictor.environment.aircraft[
                callsign
            ].requested_heading
            self.environment.aircraft[callsign].requested_fl = self.predictor.environment.aircraft[
                callsign
            ].requested_fl
            self.environment.on_route[callsign] = self.predictor.environment.on_route[callsign]

        return subtrajectories



[docs]    def observe(self, time_step: float) -> typing.Tuple[Environment, typing.Dict[str, Trajectory]]:
        """
        Evolve the Environment: apply Events and update Aircraft Trajectories.

        Parameters
        ----------
        time_step: float
            Amount of time [sec] to evolve the Environment for.

        Returns
        ----------
        Tuple[Environment, Dict[str, Trajectory]]
            Return observation of the new state: the updated Environment and the subtrajectories travelled.
        """

        # to ensure that we start moving aircraft at the correct time, keep track of any new ones
        # that may be added
        upcoming_aircaft_callsigns = [
            e.callsign for e in self.event_handler.upcoming.values() if isinstance(e, AddAircraftEvent)
        ]

        # enact the events
        self.environment = self.event_handler.forward(self.environment, time_step)

        # track any newly-added aircraft events
        newly_added_aircraft_events = {
            e.callsign: e.start_time for e in self.event_handler.completed.values()
            if isinstance(e, AddAircraftEvent) and (e.callsign in upcoming_aircaft_callsigns)
        }

        # gather the callsigns to be moved
        aircraft = np.array(list(self.environment.aircraft.keys()))

        # gather the time at which movement should start. note that if the aircraft has not
        # just been added then it will be the current env time, if not it will be the
        # time the aircraft is meant to start moving. this deals with the case where the
        # aircraft has to start moving after env.time but before env.time + time_step
        aircraft_start_times = np.array([
            self.environment.time
            if callsign not in newly_added_aircraft_events else newly_added_aircraft_events[callsign]
            for callsign in aircraft
        ])

        # get the unique start times and sort
        sorted_unique_start_times = np.sort(np.unique(aircraft_start_times))

        # storage for the subtrajectories of all aircraft
        subtrajectories = {}

        # the time the environment will have after being evolved
        env_end_time = self.environment.time + time_step

        # move the aircraft in batches, where a batch of aircraft all share the same start time
        for start_time in sorted_unique_start_times:
            # move along the environment's time to the aircraft's start time
            self.environment.time = start_time

            # get the aircraft starting at this time
            starting_aircraft = aircraft[aircraft_start_times == start_time]

            # calculate the time they should travel for
            travel_time = env_end_time - start_time

            # edge case: the aircraft wants to move at the time env.time + step_time,
            # in this case nothing needs to be done
            if travel_time == 0:
                continue

            # move along the corresponding aircraft
            stj = self.evolve_aircraft(starting_aircraft, travel_time)
            subtrajectories.update(stj)

        # finally, move the environment time to the end time
        self.environment.time = env_end_time

        # empty all the actions to apply to aircraft
        self.actions_to_apply = []

        return self.environment, subtrajectories



[docs]    def receive_actions(self, actions: typing.List[Action]):
        """
        Store Actions to apply when self.observe() is called. Actions issued to Aircraft that cannot be
        controlled by an Agent are discarded.

        Parameters
        ----------
        actions: List[Action]
            List of Actions to issue to Aircraft.
        """

        allowed_actions = [action for action in actions if self.environment.aircraft[action.callsign].controllable]
        self.actions_to_apply = allowed_actions
