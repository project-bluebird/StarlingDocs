import typing

from ..core import Environment, Reward, Trajectory
from .objective import Objective



[docs]class PreferredFlightLevel(Objective):
    """
    Quantify how close the flight level of the aircraft is to the preferred flight level (PFL).
    When the aircraft wants to descent, it should descend as late as possible. When the aircraft
    wants to ascend it should ascend as early as possible. Therefore we define PFL as the maximum
    between the sector entry level and the sector exit level.
    """


[docs]    def __init__(self, exponent: float = 2.0):
        """
        Construct a new instance.

        Parameters
        ----------
        exponent: Float = 2.0
            Exponent used in the penalty calculation
        """

        if exponent < 1.0:
            raise ValueError("Exponent should be greater or equal to 1.0.")

        self.exponent = exponent



[docs]    def evaluate(self, environment: Environment, trajectories: typing.Dict[str, Trajectory]) -> Reward:
        """
        Evaluate the penalty for deviating from the PFL at this time step. The penalty is equal to
        (abs(PFL - actual_FL) ** exp) * dt where actual_FL is the flight level at the latest trajectory control
        point, exp is a configurable exponent and dt is the time passed between the first and last control points
        in the trajectory. This is an approximation which ignores intermediate control points in the trajectory.

        Parameters
        ----------
        environment : Environment
            An Environment instance
        trajectories : Trajectory
            A dictionary with (callsign, Trajectory) items
        Returns
        ----------
        Reward
            The calculated value of the objective
        """

        delta_fl = 0.0
        for (callsign, aircraft) in environment.aircraft.items():
            if aircraft.flight_plan.entry.time <= environment.time:
                PFL = max(
                    aircraft.flight_plan.entry.fl,
                    aircraft.flight_plan.exit.fl,
                )
                dt = trajectories[callsign].control_points[-1].time - trajectories[callsign].control_points[0].time
                delta_fl += abs(PFL - trajectories[callsign].control_points[-1].fl) ** self.exponent * dt

        return Reward(-delta_fl)
