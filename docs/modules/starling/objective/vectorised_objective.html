[docs]    def align_rewards(
        self, time_values: List[np.ndarray], reward_values: List[np.ndarray], resolution: float
    ) -> np.ndarray:
        """
        Align the reward vectors for individual aircraft to create a global reward vector,
        spanning the total time these aircraft are active within the given Environment.
        It is assumed the fractional component of every time value is zero.

        Parameters
        ----------
        time_values : List[np.ndarray]
            A list of numpy arrays providing the times of trajectory control points
            for various aircraft.
        reward_values : List[np.ndarray]
            A list of numpy arrays providing the reward values calculated by
            the objective, one for each time-step (or control point), for various aircraft.
        resolution : float
            A float providing the resolution of the global reward vector in seconds.
            The resolution should be chosen such that all relevant time values are captured.

        Returns
        -------
        np.ndarray
            A global vector of reward values.
        """

        # Determine the global minimum and maximum times.
        min_time = min([t[0] for t in time_values])
        max_time = max([t[-1] for t in time_values])

        if not (min_time.is_integer() and max_time.is_integer() and resolution.is_integer()):
            raise ValueError("Fractional component of min_time, max_time and/or resolution is not zero.")

        # Create global time and reward vectors.
        time_vector = np.array(range(int(min_time), int(max_time) + 1, int(resolution)), dtype=np.float64)
        reward_vector = np.zeros(len(time_vector))

        # Add the reward values calculated for individual aircraft to the global
        # reward vector at the times for which they were calculated.
        for t_vals, r_vals in zip(time_values, reward_values):
            mask = np.any(time_vector == t_vals[:, None], axis=0)

            if sum(mask) != len(t_vals):
                raise ValueError("Resolution chosen does not include all relevant time values.")

            reward_vector[mask] += r_vals

        return reward_vector
