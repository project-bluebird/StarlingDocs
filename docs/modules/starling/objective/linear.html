[docs]    def __init__(self, weighted_objectives: typing.List[typing.Tuple[float, Objective]]):
        """
        Construct a new instance.

        Parameters
        ----------
        weighted_objectives : List[Tuple(float, Objective)]
            A list of tuples, where each tuple contains an objective and its weight.
        """

        if len(weighted_objectives) < 1:
            raise ValueError("There must be at least one Objective.")

        total_weight = 0.0
        for weight, objective in weighted_objectives:
            if weight < 0:
                raise ValueError("Weights passed to Linear metric need to be non-negative.")
            total_weight += weight

        self.weighted_objectives = []
        for weight, objective in weighted_objectives:
            self.weighted_objectives.append((weight / total_weight, objective))



[docs]    def score(self, environment: Environment, trajectories: typing.Dict[str, Trajectory]) -> Reward:
        """
        Evaluate a dictionary of Trajectories, given the Environment state,
        and determine an overall Reward to return.

        Parameters
        ----------
        environment : Environment
            An Environment instance
        trajectories : Trajectory
            A dictionary with (callsign, Trajectory) items
        Returns
        ----------
        Reward
            The calculated value of the objective
        """

        total = 0.0
        for weight, objective in self.weighted_objectives:
            total += objective.evaluate(environment, trajectories).value * weight

        return Reward(total)
