[docs]    def __init__(self, distance_threshold: float = 5.0):
        """
        Construct a new instance.

        Parameters
        ----------
        distance_threshold: float, default = 5.0
            Minimum separation threshold - distances below that are considered violations
        """

        if distance_threshold <= 0.0:
            raise ValueError("Minimum separation distance must be positive.")

        self.distance_threshold = distance_threshold



[docs]    def evaluate(self, environment: Environment, trajectories: typing.Dict[str, Trajectory]) -> Reward:
        """
        Evaluate the separation between all combinations of trajectories and return 0.0 if everything
        is OK, -1.0 if any separation threshold is violated.

        Parameters
        ----------
        environment : Environment
            An Environment instance
        trajectories : Trajectory
            A dictionary with (callsign, Trajectory) items

        Returns
        ----------
        Reward
            A Reward instance, containing the calculated value of the objective
        """

        if len(trajectories) < 2:
            return Reward(0.0)

        for (trajectory0, trajectory1) in combinations(trajectories.values(), 2):
            if trajectory0.min_distance(trajectory1) < self.distance_threshold:
                return Reward(-1.0)

        return Reward(0.0)

[docs]    def evaluate(self, environment: Environment, trajectories: typing.List[Trajectory]) -> typing.List[Reward]:
        """
        Evaluate the separation between all combinations of trajectories and return a list
        of floats. The list contains 0.0 for pairs that are OK and -1.0 for pairs where the
        separation threshold is violated.

        Parameters
        ----------
        environment : Environment
            An Environment instance
        trajectories : Trajectory
            A list of Trajectory instances

        Returns
        ----------
        list[Reward]
            The calculated values of the objective for all pair of trajectories
        """

        if len(trajectories) < 2:
            return [Reward(0.0)]

        sep = []
        for (trajectory0, trajectory1) in combinations(trajectories, 2):
            if trajectory0.min_distance(trajectory1) < self.distance_threshold:
                sep.append(Reward(-1.0))
            else:
                sep.append(Reward(0.0))

        return sep
