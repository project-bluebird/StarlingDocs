import typing
import warnings

import numpy as np

from starling.core import Environment, Reward, Trajectory
from starling.objective import Objective
from starling.core import Pos2D
from shapely.geometry import Point, LineString
from shapely.geometry.multipoint import MultiPoint



[docs]class Coordination(Objective):
    """
    Quantify how close to the coordinated exit point the aircraft exits the sector. Both
    vertical (FL) and lateral distance are considered.
    """


[docs]    def __init__(self, exponent_fl: float = 2.0, exponent_lat: float = 2.0, epsilon: float = 1e-10):
        """
        Construct a new instance.
        Parameters
        ----------
        exponent_fl: Float = 2.0
            Exponent of the flight level deviation penalty at exit
        exponent_lat: Float = 2.0
            Exponent for lateral deviation penalty at exit
        epsilon: Float = 1e-10
            Tolerance when checking if point is on boundary
        """

        if exponent_fl <= 1.0:
            raise ValueError("Exponent for flight level should be greater than 1.0.")

        if exponent_lat <= 1.0:
            raise ValueError("Exponent for lateral deviation should be greater than 1.0.")

        if epsilon <= 0.0:
            raise ValueError("Epsilon should be greater than 0.0.")

        self.exponent_fl = exponent_fl
        self.exponent_lat = exponent_lat
        self.epsilon = epsilon



[docs]    def evaluate(self, environment: Environment, trajectories: typing.Dict[str, Trajectory]) -> Reward:
        """
        Evaluate the penalty for deviating from the coordinated exit point. Any vertical difference is exponentially
        penalised. Any lateral difference above 5nm is exponentially penalised. This objective takes non-zero values
        only when at least one aircraft has exited the "outer boundary" of the sector (see Starling issue #148 for
        definition) and its passed trajectory intersects with the outer boundary. Any aircraft whose trajectories
        do not intersect with the outer boundary are given zero penalty (with a warning issued).

        Parameters
        ----------
        environment : Environment
            An Environment instance
        trajectories : Trajectory
            A dictionary with (callsign, Trajectory) items
        Returns
        ----------
        Reward
            The calculated value of the objective
        """

        if environment.airspace.outer_boundary is None:
            raise ValueError(
                "The airspace in the provided environment does not have an outer boundary, "
                "which is needed for this objective to be calculated. Outer boundaries are only "
                "defined in X, Y and I sectors at the moment and only when the airspace contain one"
                "of those sectors and no other sector."
            )

        penalty = 0.0
        for (callsign, aircraft) in environment.aircraft.items():
            if aircraft.flight_plan.entry.time <= environment.time:
                # get intersection point(s) between outer boundary lines (LineString) and trajectory line
                intersections, trajectory_line_2d = environment.airspace.outer_boundary.intersection(
                    trajectories[callsign]
                )

                if intersections.is_empty:
                    # when trajectory is fully within boundary or fully outside
                    # boundary, skip the penalty calculation for this aircraft (i.e. zero penalty)
                    warnings.warn(
                        f"Trajectory for aircraft {callsign} does not intersect with the sector's "
                        f"outer boundary. This can happen when the aircraft trajectory is entirely "
                        f"within the sector outer boundary (which means the aircraft has not "
                        f"exited the sector yet) OR when the trajectory is entirely outside the "
                        f"sector outer boundary (which means the aircraft has exited the sector "
                        f"on a previous trajectory or has not entered the sector yet).",
                        UserWarning,
                    )
                    continue

                elif type(intersections) in [Point, MultiPoint]:
                    # if there are intersections, check trajectory vs. boundary
                    p_list = trajectory_line_2d.coords
                    if all([environment.airspace.outer_boundary.boundary.contains(Point(p)) for p in p_list]):
                        # if trajectory is all inside boundary and touches boundary at one or
                        # more points, do not add penalty for this AC
                        warnings.warn(
                            f"Trajectory for aircraft {callsign} touches the sector "
                            f"outer boundary at one or more points but the aircraft "
                            f"does not exit the boundary.",
                            UserWarning,
                        )
                        continue
                    else:
                        # if at least one point is outside boundary, check for sequence of points where first
                        # point is inside boundary and second point is outside boundary and take the intersection
                        # between them as the exit point - later entry/exit points are ignored
                        # this includes case where trajectory is outside boundary and only touches it at one or more
                        # points without entering fully - exit point is the touch point
                        exit_point_2d = None
                        for i in range(len(p_list) - 1):
                            if (
                                not environment.airspace.outer_boundary.boundary.contains(Point(p_list[i + 1]))
                                and not environment.airspace.outer_boundary.boundary.boundary.distance(
                                    Point(p_list[i + 1])
                                )
                                < self.epsilon
                            ):
                                if (
                                    environment.airspace.outer_boundary.boundary.boundary.distance(Point(p_list[i]))
                                    < self.epsilon
                                ):
                                    exit_point_2d = Point(p_list[i])
                                    break
                                elif environment.airspace.outer_boundary.boundary.contains(Point(p_list[i])):
                                    exit_point_2d = environment.airspace.outer_boundary.boundary.boundary.intersection(
                                        LineString([Point(p_list[i]), Point(p_list[i + 1])])
                                    )
                                    break
                        if exit_point_2d is None:
                            # if no exit sequence is found, the intersection is an entry sequence so
                            # do not calculate any penalties for this AC
                            warnings.warn(
                                f"During the given trajectory, aircraft {callsign} enters the sector "
                                f"outer boundary but does not exit.",
                                UserWarning,
                            )
                            continue

                # find exit flight level by finding the flight level of the
                # closest trajectory control point to the exit point
                exit_fl = (
                    trajectories[callsign]
                    .control_points[np.argmin([Point(t).distance(exit_point_2d) for t in trajectory_line_2d.coords])]
                    .fl
                )

                # calculate and add penalty for difference in flight level at exit
                penalty -= (
                    abs(exit_fl - aircraft.flight_plan.exit_pos3d(environment.airspace.fixes).fl) ** self.exponent_fl
                )

                # calculate and add penalty for lateral distance at exit
                penalty -= (
                    Pos2D(exit_point_2d.y, exit_point_2d.x).distance(
                        aircraft.flight_plan.exit_pos3d(environment.airspace.fixes).pos2d()
                    )
                    ** self.exponent_lat
                )

        return Reward(penalty)
