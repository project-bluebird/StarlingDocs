import typing
import warnings

import numpy as np

from ..core import Environment, Trajectory, VectorisedReward

[docs]class VectorisedPreferredFlightLevel(VectorisedObjective):
    """
    Quantify how close the flight level of the aircraft is to the preferred flight level (PFL).
    When the aircraft wants to descend, it should descend as late as possible. When the aircraft
    wants to ascend, it should ascend as early as possible. Therefore, we define PFL as the maximum
    of the sector entry level and the sector exit level.

    See Also
    --------
    PreferredFlightLevel :
        Provides a single value instead of a vector of values, one for each time-step,
        calculated by the objective.
    """


[docs]    def __init__(self, exponent: float = 2.0):
        """
        Construct a new instance.

        Parameters
        ----------
        exponent : float, default=2.0
            Exponent used in the penalty calculation.

        Raises
        ------
        ValueError
            If the exponent is less than 1.0.
        """

        if exponent < 1.0:
            raise ValueError("Exponent should be greater than or equal to 1.0.")

        self.exponent = exponent



[docs]    def evaluate(
        self, environment: Environment, trajectories: typing.Dict[str, Trajectory], resolution: float = 1.0
    ) -> VectorisedReward:
        """
        Evaluate the penalty for deviating from the PFL at each time-step. The penalty (for each aircraft) is equal to
        (abs(PFL - actual_FL) ** exp) * dt, where actual_FL is the flight level at the latest trajectory control point
        (relative to the time-step), exp is a configurable exponent, and dt is the time passed between the control points
        in the trajectory. As dt acts as a normalising factor, it is assumed that the time between control points is constant.

        Parameters
        ----------
        environment : Environment
            An Environment instance.
        trajectories : Dict[str, Trajectory]
            A dictionary with (callsign, Trajectory) items.
        resolution : float (default=1.0)
            A float providing the resolution of the global reward vector in seconds.

        Returns
        -------
        VectorisedReward
            A vector of values, one for each time-step, calculated by the objective.
        """

        time_values = []
        reward_values = []

        for (callsign, aircraft) in environment.aircraft.items():
            if aircraft.flight_plan.entry.time <= environment.time:
                cp_locs = trajectories[callsign].locations
                time_values.append(cp_locs[:, -1])

                # Calculate the reward vector for the aircraft.
                PFL = max(aircraft.flight_plan.entry.fl, aircraft.flight_plan.exit.fl)
                dt = cp_locs[-1, -1] - cp_locs[-2, -1]
                reward_values.append(abs(PFL - cp_locs[:, -2]) ** self.exponent * dt)

        if len(reward_values) == 0:
            # Return an empty array if no aircraft were evaluated.
            warnings.warn("No aircraft were evaluated - np.ndarray of size zero returned.")
            return VectorisedReward(np.empty(shape=(0,)))
        elif len(reward_values) > 1:
            # Align the reward vectors for individual aircraft, if need be.
            reward_vector = self.align_rewards(time_values, reward_values, resolution)
            return VectorisedReward(-reward_vector)

        return VectorisedReward(-reward_values[0])
