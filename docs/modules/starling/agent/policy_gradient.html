[docs]    def __init__(
        self,
        hidden_layers: list = [2, 128, 64, 5],
        hidden_dropouts: list = [0.6, 0.6, 0.6],
        hidden_activations: list = [nn.ReLU(), nn.ReLU(), None],
        onpolicy_reset: bool = True,
        train_mode: bool = True,
        add_optim: bool = True,
        add_sched: bool = True,
    ):
        """Initialize the agent"""

        super(PolicyGradient, self).__init__()

        # Define self.model
        # Note that the first layer is the input layer (state)
        # and the last layer is the output layer (action)
        layers = []
        for i in range(len(hidden_layers) - 1):
            layers.append(nn.Linear(hidden_layers[i], hidden_layers[i + 1]))
            if hidden_dropouts[i] > 0:
                layers.append(nn.Dropout(p=hidden_dropouts[i]))
            if hidden_activations[i] is not None:
                layers.append(hidden_activations[i])

        self.model = nn.Sequential(*layers)

        # Set variables
        self.eps = np.finfo(np.float32).eps.item()
        self.action_sent = False
        self.add_optim = add_optim
        self.add_sched = add_sched

        if onpolicy_reset:
            self.onpolicy_reset()

        if train_mode:
            self.train()

        if self.add_optim:
            self.add_optimizer()

        if self.add_sched:
            self.add_scheduler()


    def add_optimizer(self, lr=0.003):
        self.optimizer = optim.Adam(self.parameters(), lr=lr)

    def add_scheduler(self, step_size=10, sc_gamma=0.5):
        self.scheduler = optim.lr_scheduler.StepLR(self.optimizer, step_size=step_size, gamma=sc_gamma)

    def onpolicy_reset(self):
        self.log_probs = []
        self.rewards = []


[docs]    def forward(self, x, use_softmax=True):
        """
        Forward pass of the network
        """
        pdparam = self.model(x)
        if use_softmax:
            pdparam = F.softmax(pdparam, dim=0)
        return pdparam


    def state_reader(self, state, plane_list=["AIR0", "AIR1"]):

        inp_state = {}

        # distance
        dist_matrix = state.dist_matrix
        if (dist_matrix is None) or (dist_matrix.size < 4):
            dist = -1000
        else:
            dist = dist_matrix[0, 1]

        inp_state["distance"] = dist

        # heading
        try:
            h1 = state.aircraft[plane_list[0]].heading
            h2 = state.aircraft[plane_list[1]].heading
            dh = (h1 - h2) % 360
        except KeyError:
            dh = -180
        except IndexError:
            dh = -180

        inp_state["heading"] = dh

        return inp_state

    def state_encoder(self, state, plane_list=["AIR0", "AIR1"]):

        inp_state = self.state_reader(state, plane_list)

        assert "distance" in inp_state.keys(), "distance not in inp_state"
        assert "heading" in inp_state.keys(), "heading not in inp_state"

        return np.array([inp_state["distance"], inp_state["heading"]])

    def find_closest_pair(self, dist_matrix, aircraft):
        if len(aircraft) < 2:
            return None

        ilt = np.tril_indices(dist_matrix.shape[0], k=0)
        dist_matrix[ilt] = np.inf
        dist_matrix = dist_matrix.astype(np.float32)
        indx2min = np.unravel_index(np.argmin(dist_matrix, axis=None), dist_matrix.shape)
        plane_list = [aircraft[indx2min[0]], aircraft[indx2min[1]]]
        return plane_list


[docs]    def generate_actions(self, raw_state, plane_mode="find_closest_pair"):
        aircraft, dist_matrix = raw_state.active_aircraft_lateral_distance()
        if len(aircraft) < 2:
            self.action_sent = False
            return []
        elif plane_mode == "find_closest_pair":
            plane_list = self.find_closest_pair(dist_matrix, aircraft)
            self.action_sent = True
            return self.generate_actions_two_planes(raw_state, plane_list=plane_list, decode_action=True)
        else:
            raise NotImplementedError


    def generate_actions_two_planes(self, raw_state, plane_list=["AIR0", "AIR1"], decode_action=True):

        state = self.state_encoder(raw_state, plane_list)
        x = torch.from_numpy(state.astype(np.float32))
        pdparam = self.forward(x)
        # pd = Categorical(logits=pdparam)
        pd = Categorical(probs=pdparam)
        action_from_agent = pd.sample()
        log_prob = pd.log_prob(action_from_agent)
        self.log_probs.append(log_prob)

        action_from_agent = action_from_agent.item()

        if decode_action:
            return self.action_decoder(action_from_agent, plane_list)
        else:
            return action_from_agent

    def action_decoder(self, action_from_agent, plane_list=["AIR0", "AIR1"]):

        action = []

        if action_from_agent == 0:
            action = []
        elif action_from_agent <= 2:
            action2send = Action(plane_list[0], "change_heading_by", int(action_from_agent) * 10)
            action2send.status = "requested"
            action.append(action2send)
        else:
            action2send = Action(plane_list[1], "change_heading_by", (int(action_from_agent) - 2) * 10)
            action2send.status = "requested"
            action.append(action2send)

        return action

    def model_train(self, gamma=0.95, onpolicy_reset=True):

        R = 0
        policy_loss = []
        rets = []
        for r in self.rewards[::-1]:
            R = r + gamma * R
            rets.insert(0, R)

        rets = torch.tensor(rets)
        # rets = (rets - rets.mean()) / (rets.std() + eps)
        for log_prob, R in zip(self.log_probs, rets):
            policy_loss.append(-log_prob * R)

        self.optimizer.zero_grad()

        policy_loss = torch.stack(policy_loss).sum()

        policy_loss.backward()
        self.optimizer.step()
        if self.add_sched:
            self.scheduler.step()

        self.optimizer.zero_grad()
        del self.log_probs[:]

        total_reward = sum(self.rewards)

        if onpolicy_reset:
            self.onpolicy_reset()

        return policy_loss, total_reward

    def save(self, path):
        os.makedirs(os.path.dirname(path), exist_ok=True)
        torch.save(self, path)

    @classmethod
    def load(self, path, eval=False):
        self = torch.load(path)
        if eval:
            self.eval()
        return self

    def add_reward(self, reward):
        if self.action_sent:
            self.rewards.append(reward)
