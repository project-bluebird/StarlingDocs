[docs]    def __init__(self, num_actions: int = 1, action_probability: float = 0.5):
        """
        Construct a new instance.
        """

        if num_actions < 0:
            raise ValueError("Number of generated actions must be non-negative.")

        self.num_actions = num_actions
        self.action_probability = action_probability



[docs]    def create_action_set(self, callsign):
        """
        Create a set of actions for a given aircraft (callsign)
        """
        action_set = [
            Action(callsign, "change_heading_by", random.choice([-30, -20, 20, 30])),
            Action(callsign, "change_flight_level_to", random.choice(np.arange(100, 400, 10))),
            Action(callsign, "on_route", 0),
            # Action(callsign, "set_route", random.choice(["AIR", "WATER"]))
        ]
        return action_set



[docs]    def generate_actions(self, environment: Environment) -> typing.List[Action]:
        """
        Randomly select planes and send random commands to them.
        This agent does not consider any previously sent commands
        """

        actions = []

        aircraft = environment.controllable_aircraft()

        if len(aircraft) > 0:
            selected_aircraft = random.choices(aircraft, k=self.num_actions)

            for ac in selected_aircraft:
                if np.random.uniform() < self.action_probability:
                    action_set = self.create_action_set(ac)
                    action = random.choice(action_set)
                    actions.append(action)

        return actions
