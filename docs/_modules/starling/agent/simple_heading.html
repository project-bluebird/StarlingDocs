[docs]    def generate_actions(self, environment: Environment) -> typing.DefaultDict:
        """
        Compute the distance matrix, find those that are too close, issue a "change_heading_*" command.
        """

        actions = []

        aircraft, dist_matrix = environment.active_aircraft_lateral_distance()

        if dist_matrix is None:
            return actions

        commanded_acs = []
        for i, ac in enumerate(aircraft):
            for j in range(i + 1, len(aircraft)):

                if (dist_matrix[i][j] <= self.min_horz_distance) and (aircraft[j] not in commanded_acs):
                    ac_action = Action(
                        aircraft[j],
                        "change_heading_to",
                        environment.aircraft[ac].heading + 90.0 % 360,
                    )
                else:
                    ac_action = Action(aircraft[j], "on_route", 0)

                actions.append(ac_action)
                commanded_acs.extend([ac, aircraft[j]])

        return actions
