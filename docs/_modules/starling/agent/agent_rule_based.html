    """
    Returns a random time from a poisson distribution with the provided lambda
    """
    rg = default_rng()
    interval = rg.poisson(p_lambda)

[docs]def get_shortest_distance_to_route(ac, environment):
    """
    Returns distance of an aircraft from a great circle between entry and exit fix
    """
    ac_pos = Pos2D(ac.lat, ac.lon)
    ac_entry_pos = list(environment.airspace.route_entry_fix(ac.flight_plan.route).values())[0]
    ac_exit_pos = list(environment.airspace.route_exit_fix(ac.flight_plan.route).values())[0]
    ac_bearing = ac_exit_pos.bearing_to(ac_pos)
    entry_bearing = ac_exit_pos.bearing_to(ac_entry_pos)
    mirror_pos = geometry.rotate_about(ac_pos, (ac_bearing - entry_bearing)*2, ac_exit_pos)
    intersect = geometry.intersection((ac_entry_pos, ac_exit_pos), (ac_pos, mirror_pos))
    dist = intersect.distance(ac_pos)
    return dist




[docs]def phase_cat(ac_afl, ac_xfl):
    """
    Returns the flight phase of an aircraft given its current flight level and exit flight level
    """
    if ac_xfl > ac_afl:
        return "outbound"
    elif ac_xfl == ac_afl:
        return "overflight"
    else:
        return "inbound"




[docs]def vertical_interaction_exists(ac1_afl, ac1_xfl, ac2_afl, ac2_xfl):
    """
    Returns whether a vertical interaction exists between two aircraft given their current and required exit levels

    Used to find planning problems, should be used alongside lateral assessments to judge if interactions are relevant
    """
    if min(ac1_afl, ac1_xfl) > max(ac2_afl, ac2_xfl) or min(ac2_afl, ac2_xfl) > max(ac1_afl, ac1_xfl):
        return False
    else:
        return True




[docs]def is_steppable(ac1_afl, ac1_xfl, ac2_afl, ac2_xfl):
    """
    Returns whether a step climb/descent is viable for two aircraft based on levels alone - no lateral judgement
    """
    if (ac1_afl > ac2_afl + 10 and ac1_xfl >= ac2_xfl + 10) or (ac2_afl > ac1_afl + 10 and ac2_xfl >= ac1_xfl):
        return True
    else:
        return False




[docs]def climb_descend_distance(start_level, end_level, ground_speed):
    """
    Returns distance required to climb/descend a certain distance based on linear model. Climb/descent rate hard coded.
    """
    height_change = start_level - end_level
    time_to_lose = height_change/(20/60) # 20 is climb/desc rate in flight levels hardcoded in linear pilot
    distance_required = time_to_lose*ground_speed/(60*60)
    return distance_required




[docs]def tod_in_range(ac_pos, ac_afl, ac_xfl, ac_spd, exit_point):
    """
    Returns whether an aircraft has passed its top of descent based on linear model. Climb/descent rate hard coded.
    """
    height_to_lose = ac_afl - ac_xfl
    time_to_lose = height_to_lose*60/20  # 20 is climb/desc rate in flight levels hardcoded in linear pilot
    distance_to_lose = time_to_lose*ac_spd/(60*60)
    distance_to_run = ac_pos.distance(exit_point)
    if distance_to_run > distance_to_lose:
        return False
    else:
        return True




[docs]def interaction_type(ac1_afl, ac1_xfl, ac2_afl, ac2_xfl):
    """
    Returns the interaction type between two aircraft as a combination of inbound, outbound and overflight
    """
    if min(ac1_afl, ac1_xfl) >= max(ac2_afl, ac2_xfl) + 10 or min(ac2_afl, ac2_xfl) >= max(ac1_afl, ac1_xfl) + 10:
        type_key = 0
    else:
        type_raw = []
        type_raw.append(phase_cat(ac1_afl, ac1_xfl))
        type_raw.append(phase_cat(ac2_afl, ac2_xfl))
        if type_raw == ["inbound", "inbound"]:
            type_key = 1
        elif type_raw == ["overflight", "overflight"]:
            type_key = 2
        elif type_raw == ["outbound", "outbound"]:
            type_key = 3
        elif type_raw.__contains__("inbound") and type_raw.__contains__("overflight"):
            type_key = 4
        elif type_raw.__contains__("inbound") and type_raw.__contains__("outbound"):
            type_key = 5
        elif type_raw.__contains__("overflight") and type_raw.__contains__("outbound"):
            type_key = 6
        else:
            raise Exception("error in interaction classification")

    return type_key




[docs]def choose_plan(ac1, ac2, airspace):
    """
    Returns one of three plans for solving a simple I-sector head-on scenario, given the airspace and two aircraft
    """
    interact_type = interaction_type(ac1.fl, ac1.flight_plan.exit.fl, ac2.fl, ac2.flight_plan.exit.fl)
    ac1_exit_pos = list(airspace.route_exit_fix(ac1.flight_plan.route).values())[0]
    ac2_exit_pos = list(airspace.route_exit_fix(ac2.flight_plan.route).values())[0]
    ac1_pos = Pos2D(ac1.lat, ac1.lon)
    ac2_pos = Pos2D(ac2.lat, ac2.lon)
    if interact_type == 0:  # No interaction
        plan = "independent"
    elif interact_type == 1:  # In/In
        if is_steppable(ac1.fl, ac1.flight_plan.exit.fl, ac2.fl, ac2.flight_plan.exit.fl):
            plan = "step"
        else:
            plan = "split"
    elif interact_type == 2:  # Over/Over
        plan = "split"
    elif interact_type == 3:  # Out/Out
        if is_steppable(ac1.fl, ac1.flight_plan.exit.fl, ac2.fl, ac2.flight_plan.exit.fl):
            plan = "step"
        else:
            plan = "split"
    elif interact_type == 4:  # In/Over
        if phase_cat(ac1.fl, ac1.flight_plan.exit.fl) == "inbound":
            start_level = ac2.fl + 10
            target_level = ac1.flight_plan.exit.fl
            speed = ac1.speed
            exit_dist = ac1_pos.distance(ac1_exit_pos)
        else:
            start_level = ac1.fl + 10
            target_level = ac2.flight_plan.exit.fl
            speed = ac2.speed
            exit_dist = ac2_pos.distance(ac2_exit_pos)
        time_to_exit = exit_dist / speed
        time_to_lose = (climb_descend_distance(start_level, target_level, speed) + 5) / speed  # 5 extra dist for cross
        time_to_cross = ac1_pos.distance(ac2_pos) / (ac1.speed + ac2.speed)

        if (time_to_exit - time_to_cross) > time_to_lose and ac1.requested_fl != ac2.requested_fl:
            plan = "step"
        else:
            plan = "split"
    elif interact_type == 5:  # In/Out
        plan = "split"
    elif interact_type == 6:  # Over/Out
        if phase_cat(ac1.fl, ac1.flight_plan.exit.fl) == "outbound":
            start_level = ac1.fl
            target_level = ac2.fl
            speed = ac1.speed
        else:
            start_level = ac2.fl
            target_level = ac1.fl
            speed = ac1.speed
        time_to_gain = (climb_descend_distance(target_level, start_level, speed)) / speed
        time_to_cross = (ac1_pos.distance(ac2_pos) + 5) / (ac1.speed + ac2.speed)  # +5 for full cross
        if time_to_gain > time_to_cross:
            plan = "step"
        else:
            plan = "split"
    else:
        raise Exception("Error in choose plan")
    return plan




[docs]def get_best_level_independent(ac, ac_exit_pos, ac_phase):
    """
    Returns best cleared level for a flight assuming no other aircraft interactions - early climb, late descent
    """
    ac_pos = Pos2D(ac.lat, ac.lon)
    if ac.requested_fl == ac.flight_plan.exit.fl:
        best_level = None
    elif ac_phase == "outbound":
        best_level = ac.flight_plan.exit.fl
    elif ac_phase == "inbound":
        if tod_in_range(ac_pos, ac.fl, ac.flight_plan.exit.fl, ac.speed, ac_exit_pos):
            best_level = ac.flight_plan.exit.fl
        else:
            best_level = None
    else:
        raise Exception("Error in get_best_level_independent")
    return best_level

[docs]def split_logic(ac1_callsign, ac1, ac2, ac1_exit_pos, ac2_exit_pos, split_phase, split_phases, env):
    """
    Logic to return best action and phase update for "split" plan
    """
    ac1_pos = Pos2D(ac1.lat, ac1.lon)
    ac2_pos = Pos2D(ac2.lat, ac2.lon)
    vert_int_exists = vertical_interaction_exists(ac1.fl, ac1.flight_plan.exit.fl, ac2.fl, ac2.flight_plan.exit.fl)
    action = None
    new_split_phase = split_phase
    if not vert_int_exists and split_phase != "complete":
        action = Action(ac1_callsign, "on_route", 0)
        new_split_phase = "complete"
    elif split_phase == "initial":
        action = Action(ac1_callsign, "change_heading_by", 10)
        new_split_phase = "vectoring"
    elif split_phase == "vectoring":
        if get_shortest_distance_to_route(ac1, env) > 3:
            action = Action(ac1_callsign, "change_heading_by", -10)
            new_split_phase = "parallel"
    elif split_phase == "parallel":
        ac1_home_bearing = ac1_pos.bearing_to(ac1_exit_pos)
        ac2_home_bearing = ac2_pos.bearing_to(ac2_exit_pos)
        cap = get_cap(ac1, ac2, ac1_home_bearing, ac2_home_bearing)
        if cap > 7:
            action = Action(ac1_callsign, "on_route", 0) #on_route
            new_split_phase = "complete"
    elif split_phase == "complete":
        new_split_phase = "complete"
    else:
        raise Exception("invalid split phase")

    if action is None and not split_phases.__contains__("initial"):
        best_level = get_best_level_independent(ac1, ac1_exit_pos, phase_cat(ac1.fl, ac1.flight_plan.exit.fl))
        if best_level is None:
            action = None
        else:
            action = Action(ac1_callsign, "change_flight_level_to", best_level)
    return action, new_split_phase




[docs]def get_action_step(ac1_callsign, ac1, ac2, ac1_exit_pos, ac1_phase):
    """
    Returns action for the "step" plan
    """
    ac1_pos = Pos2D(ac1.lat, ac1.lon)
    if get_cap(ac1, ac2) > 5:
        best_level = get_best_level_independent(ac1, ac1_exit_pos, ac1_phase)
    elif ac1_phase == "overflight":
        best_level = None
    elif ac1_phase == "outbound":
        if ac2.fl > ac1.fl and ac1.requested_fl < ac1.flight_plan.exit.fl:
            best_level = math.floor(min(ac2.fl, ac2.requested_fl) / 10) * 10 - 10
            if best_level > ac1.flight_plan.exit.fl:
                best_level = ac1.flight_plan.exit.fl
        else:
            best_level = get_best_level_independent(ac1, ac1_exit_pos, ac1_phase)
    else:
        if tod_in_range(ac1_pos, ac1.fl, ac1.flight_plan.exit.fl, ac1.speed, ac1_exit_pos):
            if ac2.fl < ac1.fl and ac1.requested_fl > ac1.flight_plan.exit.fl:
                best_level = math.ceil(max(ac2.fl, ac2.requested_fl) / 10) * 10 + 10
                if best_level < ac1.flight_plan.exit.fl:
                    best_level = ac1.flight_plan.exit.fl
            else:
                best_level = get_best_level_independent(ac1, ac1_exit_pos, ac1_phase)
        else:
            best_level = None
    if best_level is None or best_level == ac1.requested_fl:
        action = None
    else:
        action = Action(ac1_callsign, "change_flight_level_to", best_level)
    return action




[docs]def get_cap(ac1, ac2, ac1_hdg_override=None, ac2_hdg_override=None):
    """
    Returns closest approach for two aircraft based on straight line forecast behaviour. Accepts heading overrides.
    """
    geodesic = Geod(ellps='WGS84')
    npts = 600

    ac1_del_s = ac1.speed
    ac2_del_s = ac2.speed

    if ac1_hdg_override is None:
        ac1_az = ac1.heading
    else:
        ac1_az = ac1_hdg_override

    if ac2_hdg_override is None:
        ac2_az = ac2.heading
    else:
        ac2_az = ac2_hdg_override

    ac1_tp = geodesic.fwd_intermediate(ac1.lon, ac1.lat, ac1_az, npts=npts, del_s=ac1_del_s)
    ac2_tp = geodesic.fwd_intermediate(ac2.lon, ac2.lat, ac2_az, npts=npts, del_s=ac2_del_s)
    _, _, distance = geodesic.inv(ac1_tp.lons, ac1_tp.lats, ac2_tp.lons, ac2_tp.lats)
    return min(distance) * convert.M_TO_NMI  # convert to nm

        """
        Caches an action to be executed at a later time point

        interval = get_poisson_interval(self.poisson)
        self.action_cache = action

[docs]    def generate_actions(self, environment: Environment):
        """
        Returns actions. On first run creates a plan, and updates the plan as the situation evolves.
        """
        # Set variables for planning and controlling logic
        action = None
        ac1_callsign, ac1, ac1_exit_pos, ac1_phase = get_ac_info(environment, self.stripscan_position)
        ac2_callsign, ac2, ac2_exit_pos, ac2_phase = get_ac_info(environment, (self.stripscan_position + 1) % 2)

        # If previously cached action exists, wait and issue at correct time
        if self.action_cache is not None:
            if environment.time > self.action_init_time:
                action = [self.action_cache]
                self.action_cache = None
        else:
            # Planning
            if not self.plan:
                self.make_plan(ac1, ac2, ac1_callsign, ac2_callsign, environment.airspace)

            # Controlling

            # Make no action if current aircraft is not in sector
            if not environment.airspace.contains(ac1.pos3d()):
                action = None

            # Get action based on plan
            elif self.plan == "independent":
                action = get_action_independent(ac1_callsign, ac1, ac1_exit_pos, ac1_phase)

            elif self.plan == "split":
                if self.split_phases == ["complete", "complete"]:
                    self.plan = "independent"
                split_phase = self.split_phases[self.stripscan_position]
                action = get_action_split(ac1_callsign, ac1, ac2, ac1_exit_pos, ac2_exit_pos, split_phase, self.split_phases, environment)
                self.split_phases[self.stripscan_position] = split_phase_update(ac1_callsign, ac1, ac2, ac1_exit_pos, ac2_exit_pos, split_phase, self.split_phases, environment)

            elif self.plan == "step":
                if get_cap(ac1, ac2) > 5:
                    self.plan = "independent"
                action = get_action_step(ac1_callsign, ac1, ac2, ac1_exit_pos, ac1_phase)

            else:
                raise Exception("No plan assigned error!")

            if action is not None:
                if self.poisson > 0:
                    self.cache_action(action, environment.time)
                    action = None
                else:
                    action = [action]
        self.advance_stripscan()
        if action is None:
            action = []
        # else:
        #     print(action[0].callsign, action[0].kind, action[0].value)
        return action
