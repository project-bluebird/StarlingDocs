    def from_str(s: str) -> Contract:
        """
        Construct a new instance from string representation.
        """

        s.strip()

        s = s.split()
        fl = s[0]
        time = s[1]

        if fl[-2:] != "FL":
            raise ValueError(f"Invalid flight level string: {fl}")

        [hr, min, sec] = time.split(":")
        time = (3600.0 * float(hr)) + (60.0 * float(min)) + float(sec)

        return Contract(float(fl[:-2]), time)


    def __str__(self) -> str:
        """
        Create a human readable representation of the instance.
        """

        time_str = time.strftime("%H:%M:%S", time.gmtime(self.time))

        return f"{self.fl}FL {time_str}"
