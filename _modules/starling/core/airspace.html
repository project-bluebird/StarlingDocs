        """

        """

        return any(map(lambda sector: sector.contains(position), self.sectors.values()))



[docs]    def boundary(self) -> Area:
        """
        Return the boundary of the entire Airspace i.e., the union of all Sectors' Volumes.
        """

        volumes = list(chain.from_iterable([sector.volumes for sector in self.sectors.values()]))
        polygons = [volume.area.boundary for volume in volumes]

        poly = polygons[0]
        if len(polygons) > 1:
            for poly2 in polygons[1:]:
                poly = poly.union(poly2)
        # NOTE: Shapely returns lon,lat and it is Pos2D(lat,lon)
        coords = [Pos2D(y, x) for x, y in poly.exterior.coords]

        return Area(coords)



[docs]    def boundary_fixes(self) -> Fixes:
        """
        Return Fixes that are on the Airspace boundary.
        """

        boundary_fixes = {}
        airspace_area = self.boundary()
        for fix, pos in self.fixes.places.items():
            if airspace_area.on_boundary(pos):
                boundary_fixes[fix] = pos

        return Fixes(boundary_fixes)



[docs]    def route_boundary_fixes(self, route: Route) -> Fixes:
        """
        Return the Route Fixes that are on the Airspace boundary.
        """

        boundary_fixes = self.boundary_fixes()
        route_boundary_fixes = {fix: pos for fix, pos in boundary_fixes.places.items() if fix in route.names}

        return Fixes(route_boundary_fixes)



[docs]    def route_entry_fix(self, route: Route) -> Fixes:
        """
        Return Route entry Fix.
        """

        boundary_fixes = self.route_boundary_fixes(route)
        # get the first boundary Fix on Route
        for fix in route.names:
            if fix in boundary_fixes.places.keys():
                return {fix: boundary_fixes.places[fix]}



[docs]    def route_exit_fix(self, route: Route) -> dict[str, Pos2D]:
        """
        Return Route exit Fix.
        """

        boundary_fixes = self.route_boundary_fixes(route)
        # get the last boundary Fix on Route
        for fix in route.names[::-1]:
            if fix in boundary_fixes.places.keys():
                return {fix: boundary_fixes.places[fix]}



[docs]    def find_volume(self, position: Pos2D) -> Volume:
        """
        Return the Volume of Airspace that contains position.
        """

        for name, sector in self.sectors.items():
            for volume in sector.volumes:
                if volume.area.contains(position):
                    return volume
        return None



[docs]    def find_fl_lim(self, position: Pos2D) -> list[float, float]:
        """
        Find the min/max flight level allowed in the part of Airspace containing position.
        """

        volume = self.find_volume(position)
        if volume is not None:
            return [volume.min_fl, volume.max_fl]
        return [None, None]

[docs]    def get_geometry_collection(self) -> GeometryCollection:
        """
        Return a GeometryCollection of all Area boundaries and Fix coordinates.
        """

        polygons = []
        for sector_name in self.sectors:
            volumes = self.sectors[sector_name].volumes
            polygons.extend([volume.area.boundary for volume in volumes])
        # NOTE: Shapely uses (lon,lat)
        fixes = [Point(fix.lon, fix.lat) for fix in self.fixes.places.values()] 
        collection = GeometryCollection(fixes + polygons)

        return collection



[docs]    def update_airspace(self, collection: GeometryCollection):
        """
        Update Airspace Fix locations and Area boundaries of all Volumes from collection.
        """

        polygons = []
        fixes = []
        for geom in collection.geoms:
            if type(geom) == Polygon:
                polygons.append(geom)
            elif type(geom) == Point:
                fixes.append(geom)

        # Update Fix positions    
        for i, fix in enumerate(self.fixes.places):
            lon, lat = fixes[i].coords.xy
            self.fixes.places[fix] = Pos2D(lat[0], lon[0])

        # Update Area boundaries
        counter = 0
        for sector_name in self.sectors:
            volumes = self.sectors[sector_name].volumes
            for i, v in enumerate(volumes):
                self.sectors[sector_name].volumes[i].area.boundary = polygons[counter]
                counter += 1

        """

        collection = self.get_geometry_collection()
        rotated_collection = rotate(collection, angle)

[docs]    def translate(self, xoffset: float = 0.0, yoffset: float = 0.0):
        """
        Move Airspace by offset [nmi] in x and/or y direction.
        """

        collection = self.get_geometry_collection()
        xoffset *= convert.NMI_TO_DEG
        yoffset *= convert.NMI_TO_DEG
        translated_collection = translate(collection, xoff=xoffset, yoff=yoffset)
        self.update_airspace(translated_collection)
