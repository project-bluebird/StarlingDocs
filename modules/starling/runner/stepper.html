[docs]    def compute_dist_matrix(self):
        """Calculate distance matrix between all aircraft."""

        aircraft, dist_matrix = self.environment.active_aircraft_lateral_distance()
        if dist_matrix is not None:
            self.dm_times.append(self.istep)
            if dist_matrix.size >= 4:
                iut = np.triu_indices(dist_matrix.shape[0], k=1)
                triu_dist_matrix = dist_matrix[iut]
                self.triu_dist_matrix = triu_dist_matrix
                self.dm_vals.append(np.sum(laplace.pdf(triu_dist_matrix, scale=0.5)) / len(triu_dist_matrix))
            else:
                self.dm_vals.append(0)

        self.dist_matrix = dist_matrix

        return aircraft, dist_matrix

[docs]    def git_params(self):
        """Gets the current git commit hash and author"""
        gitstr = git.Git(self.dir_path)
        log_info_split = gitstr.log("-1").split()
        commit_index = log_info_split.index("commit") + 1
        author_index = log_info_split.index("Author:") + 1
        git_hash = log_info_split[commit_index]
        git_author = log_info_split[author_index] + " " + log_info_split[author_index + 1]
        return git_hash, git_author

[docs]    def next_trajectory(self, actions):
        """Uses the simulator (predictor) to calculate new trajectories
        and moves aircraft on these trajectories."""
        trajectories = dict()
        for callsign in self.environment.aircraft:
            trajectory = self.predictor.future_trajectory(
                callsign, self.environment, self.radar_sweep_period, actions
            )

            if trajectory:
                self.environment.aircraft[callsign].travel(trajectory)
                trajectories[callsign] = trajectory

        return trajectories

[docs]    def compute_reward(
        self,
        aircraft,
        dist_matrix,
        approx_k: int = 50,
        scale_reward: float = 1.0,
        plane_pair: typing.Union[list, None] = None,
    ):
        """Calculates reward for PG agent - will be moved to agent class"""

        if (dist_matrix is None) or (dist_matrix.size < 4):
            return 0.0

        if plane_pair is not None:
            try:
                indx1 = aircraft.index(plane_pair[0])
                indx2 = aircraft.index(plane_pair[1])
                distance_pair = dist_matrix[indx1, indx2]
            except ValueError:
                return None
        else:
            iut = np.triu_indices(dist_matrix.shape[0], k=1)
            triu_dist_matrix = dist_matrix[iut]
            distance_pair = np.min(triu_dist_matrix)

        if distance_pair < self.distance_threshold:
            approx_step = 1.0 / (1 + np.exp(-2 * approx_k * (distance_pair - self.distance_threshold * 1.01)))
            scaled_log = np.log(distance_pair / (self.distance_threshold))
            return (approx_step - 1.0 + scaled_log) * scale_reward
        else:
            return 0.0


    def is_done(self, aircraft, min_steps: int = 500):

        if not hasattr(self, "triu_dist_matrix"):
            return False
        elif np.min(self.triu_dist_matrix) < self.distance_threshold:
            return True
        elif (self.istep > min_steps) and len(aircraft) == 0:
            return True
        else:
            return False


[docs]    def step(self, actions=[]):
        """Advances simulation by one time step, calculates reward and logs."""

        # Should be removed
        aircraft, dist_matrix = self.compute_dist_matrix()

        # evolve trajectory and log environment state
        trajectories = self.next_trajectory(actions)

        # calculate and log reward
        # NOTE: should be replaced with this when reward is moved to separate class:
        # self.objective.evaluate(self.environment, trajectories)
        self.reward = self.compute_reward(aircraft, dist_matrix, scale_reward=100, plane_pair=None)

        # log action, BEFORE evolving time in state (as action is associated with previous state)
        self.log_action(actions)

        # increment time and log environment an reward
        self.environment.time += self.radar_sweep_period
        self.istep += 1
        self.log_reward(self.reward)
        self.log_environment()

        # --- STATE
        self.state = self.compute_state(**{"dist_matrix": self.dist_matrix})

        # --- DONE
        self.done = self.is_done(aircraft)

    def setup_logger(name, filename, overwrite=True):
        """Setup an INFO-level logger object which logs a message,
        as well as a handler and formatter"""

        logger = logging.getLogger(name)
        logger.setLevel(logging.INFO)
        if overwrite:
            f_handler = logging.FileHandler(filename, "w+")
        else:
            f_handler = logging.FileHandler(filename)
        f_format = logging.Formatter("%(message)s")
        f_handler.setFormatter(f_format)
        logger.handlers.clear()
        logger.addHandler(f_handler)

        return logger
