[docs]    def __init__(self, entry: Contract, exit: Contract, route: Route):
        """
        Construct a new instance.
        """

        if entry.time < 0.0:
            raise ValueError("Entry time must be non-negative.")

        if entry.time > exit.time:
            raise ValueError("Exit time must be after entry time.")

        self.entry = entry
        self.exit = exit

        self.route = route



[docs]    @staticmethod
    def from_json(s: str) -> FlightPlan:
        """
        Construct a new instance from JSON representation.
        """

        data = json.loads(s)

        entry = Contract.from_str(f"{data['entry']}")
        exit = Contract.from_str(f"{data['exit']}")
        route = Route.from_json(json.dumps(data["route"]))

        return FlightPlan(entry, exit, route)

[docs]    def exit_pos3d(self, fixes: Fixes) -> Pos3D:
        """
        Get the exit Contract as Pos3D.
        """

        # Boundary exit fix is the 2nd to last fix on Route
        fix_name = self.route.names[-2]
        fix_pos = fixes.places[fix_name]
        fl = self.exit.fl

        return fix_pos.pos3d(fl)



[docs]    def entry_pos3d(self, fixes: Fixes) -> Pos3D:
        """

        fix_name = self.route.names[1]
