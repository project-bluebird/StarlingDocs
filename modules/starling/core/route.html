[docs]    def length(self, fixes: Fixes) -> float:
        """
        Calculate the length [nmi] of the Route.
        """

        total = 0.0
        for (a, b) in self.segments():
            total += fixes.places[a].distance(fixes.places[b])

        return total



[docs]    def nearest(self, fixes: Fixes, position: Pos2D) -> tuple[Pos2D, float]:
        """
        Get the nearest on-route coordinate.
        """

        min_dist = float("inf")
        nearest = fixes.places[next(iter(self.names))]
        for n1, n2 in zip(self.names, self.names[1:]):
            s1 = fixes.places[n1]
            s2 = fixes.places[n2]

            s1_s2 = Pos2D(s2.lat - s1.lat, s2.lon - s1.lon)
            s1_p = Pos2D(position.lat - s1.lat, position.lon - s1.lon)

            s = math.sqrt((s1_s2.lat ** 2) + (s1_s2.lon ** 2))

            t = ((s1_p.lat * s1_s2.lat) + (s1_p.lon * s1_s2.lon)) / s
            t = min(max(t, 0.0), 1.0)

            q = Pos2D(s1.lat + (t * (s2.lat - s1.lat)), s1.lon + (t * (s2.lon - s1.lon)))

            dist = q.distance(position)

            if dist < min_dist:
                min_dist = dist
                nearest = q

        return (nearest, min_dist)
