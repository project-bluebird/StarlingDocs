    """
    Trajectory factory class.
    helps concisely create a Trajectory
    and an accompanying list of Actions
    which would enable an Aircraft to fly between the given control points.
    """


[docs]    def __init__(self, start, end, num_control_points, num_eval_points=3):
        """
        Construct a new instance.
        """

        if num_control_points <= 2:
            raise ValueError("There must be at least three control-points.")

        if num_eval_points < 0:
            raise ValueError("Number of evaluation points must be non-negative.")

        self.start = start
        self.end = end
        self.num_control_points = num_control_points
        self.num_eval_points = num_eval_points

        self.control_points = []

        distance = start.distance(end)
        bearing = start.bearing_to(end)
        d_fl = end.fl - start.fl
        for x in np.linspace(0.0, 1.0, num_control_points)[1 : num_control_points - 1]:
            self.control_points.append(start.forward(distance * x, bearing))
            self.control_points[-1].fl = start.fl + (d_fl * x)



[docs]    def get_control_points(self, deltas, speed):
        """
        Get the control-points offset by some deltas.
        """

        from ..core import Pos3D, Pos4D

        if len(deltas) != len(self.control_points):
            raise ValueError("Number of deltas must equal number of intermediate control-points.")

        time = 0.0
        cps = []
        cps.append(self.start.pos4d(time))

        for cp, [d_lat, d_lon, d_fl] in zip(self.control_points, deltas):
            lat = cp.lat + d_lat
            lon = cp.lon + d_lon
            fl = cp.fl + d_fl
            time += cps[-1].pos3d().distance(Pos3D(lat, lon, fl)) / speed
            cps.append(Pos4D(lat, lon, fl, time))

        time += cps[-1].pos3d().distance(self.end) / speed
        cps.append(self.end.pos4d(time))

        return cps



[docs]    def get_trajectory(self, deltas, speed):
        """
        Construct a Trajectory from a parameterisation.
        """

        from ..core import Trajectory

        if len(deltas) != len(self.control_points):
            raise ValueError("Number of deltas must equal number of intermediate control-points.")

        time = 0.0
        cps = self.get_control_points(deltas, speed)
        traj = []
        traj.append(self.start.pos4d(time))
        for (a, b) in zip(cps, cps[1:]):
            dx = a.distance(b) / (self.num_eval_points + 1)
            bearing = a.bearing_to(b)
            d_fl = b.fl - a.fl
            for x in np.linspace(0.0, 1.0, self.num_eval_points + 2)[1 : self.num_eval_points + 1]:
                traj.append(traj[-1].forward(dx, bearing, speed))
                traj[-1].fl = a.fl + (d_fl * x)
            traj.append(b)

        return Trajectory(traj)



[docs]    def get_actions(self, callsign, deltas, speed):
        """
        Construct an Action set from a parameterisation.
        """

        from ..core import Action

        if len(deltas) != len(self.control_points):
            raise ValueError("Number of deltas must equal number of intermediate control-points.")

        cps = self.get_control_points(deltas, speed)
        actions = []
        last_bearing = math.inf
        for cp_a, cp_b in zip(cps[:-1], cps[1:]):
            bearing = cp_a.bearing_to(cp_b)

            if abs(last_bearing - bearing) > 5.0:
                actions.append(Action(callsign, "change_heading_to", bearing))

            if abs(cp_a.fl - cp_b.fl) > 10.0:
                actions.append(Action(callsign, "change_flight_level_to", cp_b.fl))

        return actions
