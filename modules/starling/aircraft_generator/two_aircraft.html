[docs]class TwoAircraft(AircraftGenerator):
    """
    Two Aircraft traveling head on:
        - start at opposite ends of the same route
        - first Aircraft is an overflier
        - second Aircraft is an overflier, climber or descender
        - the Aircraft either travel on the same FL or have crossing entry/exit FL contracts
    """


[docs]    def __init__(self, total_time: float, speed_range: typing.List[float] = None):
        """
        Construct a new instance.

        Parameters
        ----------
        total_time: float
            The exit_contract time for both Aircraft. If speed_range is not provided, speed of both Aircraft
            is chosen such that the contracts would be satisfied.
        speed_range: list[float, float]
            Optional range of [min,max] speeds from which to randomly choose Aircraft speed. If not provided,
            speed of both Aircraft is set based on the total_time parameter.
        """

        if (total_time is not None) and total_time <= 0.0:
            raise ValueError("Total time must be positive.")

        if (speed_range is not None) and len(speed_range) != 2:
            raise ValueError("Speed range must be a list of length 2.")

        self.total_time = total_time
        self.speed_range = speed_range



[docs]    def get_overflier_contracts(self, route: Route, airspace: Airspace, scenario_type: str) -> typing.List[Contract]:
        """
        Create entry and exit contract for the first/overflier aircraft.

        Parameters
        ----------
        route: Route
            The Route which the Aircraft is to fly
        airspace: Airspace
            The Airspace the Aircraft is flying through
        scenario_type: str
            One of "overflier", "climber" or "descender". Describes the behaviour of the second Aircraft.

        Returns
        ----------
        list[Contract]
            The [entry, exit] contract for the overflier aircraft
        """

        entry_FLs = airspace.get_FLs_entry(route)
        exit_FLs = airspace.get_FLs_exit(route)

        # find FLs that are allowed at both entry and exit
        constant_FLs = list(set.intersection(set(entry_FLs), set(exit_FLs)))
        constant_FLs.sort()
        contract_FL = random.choice(constant_FLs)

        # make sure climber/descender can start below/above overflier
        # - leave lowest/highest FL band clear
        if scenario_type == "descender":
            contract_FL = min(contract_FL, constant_FLs[-2])
        elif scenario_type == "climber":
            contract_FL = max(contract_FL, constant_FLs[1])

        return [Contract(contract_FL, 0.0), Contract(contract_FL, self.total_time)]



[docs]    def get_descender_contracts(self, overflier_fl: int, route: Route, airspace: Airspace) -> typing.List[Contract]:
        """
        Create entry and exit contract for descender aircraft.

        Parameters
        ----------
        route: Route
            The Route which the Aircraft is to fly
        airspace: Airspace
            The Airspace the Aircraft is flying through

        Returns
        ----------
        list[Contract]
            The [entry, exit] contract for the descender aircraft
        """

        entry_FLs = airspace.get_FLs_entry(route)
        exit_FLs = airspace.get_FLs_exit(route)

        # start above the overflier (up to the highest allowed entry FL limit)
        start_fl = random.choice(np.arange(overflier_fl + 10, entry_FLs[-1] + 10, 10))

        # exit at or below the overflier FL to ensure paths cross
        exit_fl = random.choice(np.arange(exit_FLs[0], overflier_fl + 10, 10))

        return [Contract(start_fl, 0.0), Contract(exit_fl, self.total_time)]



[docs]    def get_climber_contracts(self, overflier_fl: int, route: Route, airspace: Airspace) -> typing.List[Contract]:
        """
        Create entry and exit contract for climber aircraft.

        Parameters
        ----------
        route: Route
            The Route which the Aircraft is to fly
        airspace: Airspace
            The Airspace the Aircraft is flying through

        Returns
        ----------
        list[Contract]
            The [entry, exit] contract for the climber aircraft
        """

        entry_FLs = airspace.get_FLs_entry(route)
        exit_FLs = airspace.get_FLs_exit(route)

        # start below the overflier (down to the lowest allowed entry FL)
        start_fl = random.choice(np.arange(entry_FLs[0], overflier_fl, 10))

        # climb to FL at or above overflier FL to ensure paths cross
        exit_fl = random.choice(np.arange(overflier_fl, exit_FLs[-1] + 10, 10))

        return [Contract(start_fl, 0.0), Contract(exit_fl, self.total_time)]



[docs]    def generate_aircraft(
        self, airspace: Airspace, routes: typing.List[Route], scenario_type: str = "random"
    ) -> typing.Dict[str, Aircraft]:
        """
        Generate two Aircraft for the given Airspace.

        Parameters
        ----------
        airspace: Airspace
            The Airspace the Aircraft is flying through
        routes: list[Route]
            The available Routes in the Airspace (choose one at random for the Aircraft Route).
        scenario_type: str
            One of "random", "overflier", "climber" or "descender". If "random" - choose one of ["overflier", "climber", "descender"].
            Describes the behaviour of the second Aircraft.

        Returns
        ----------
        Dict[str, Aircraft]
            Two Aircraft travelling in opposite direction on the same Route, both either on the same FL or with paths crossing.
        """

        aircraft = {}

        # set or randomly select scenario type
        if scenario_type == "random":
            aircraft_scenario = random.choice(["overflier", "climber", "descender"])
        else:
            aircraft_scenario = scenario_type

        route_fwd = random.choice(routes)
        route_rev = Route(route_fwd.names[::-1])
        routes = [route_fwd, route_rev]

        # randomly generate entry/exit contracts for first aircraft
        contracts_fwd = self.get_overflier_contracts(route_fwd, airspace, aircraft_scenario)
        overflier_fl = contracts_fwd[0].fl

        # generate entry/exit contracts for second aircraft
        if aircraft_scenario == "overflier":
            contracts_rev = contracts_fwd
        elif aircraft_scenario == "climber":
            contracts_rev = self.get_climber_contracts(overflier_fl, route_rev, airspace)
        elif aircraft_scenario == "descender":
            contracts_rev = self.get_descender_contracts(overflier_fl, route_rev, airspace)
        contracts = [contracts_fwd, contracts_rev]

        vertical_speed = 0

        for i in range(2):

            callsign = f"AIR{i}"

            # speed is in knots i.e., nmi per hour
            if self.speed_range is None:
                speed = route_fwd.length(airspace.fixes) / (self.total_time / 3600.0)
            else:
                speed = self.speed_range[0] + (self.speed_range[1] - self.speed_range[0]) * np.random.uniform()

            route = routes[i]
            entry, exit = contracts[i]
            flight_plan = FlightPlan(entry, exit, route)

            fix1 = airspace.fixes.places[route.names[0]]
            heading = fix1.bearing_to(airspace.fixes.places[route.names[1]])

            pos = fix1.pos3d(entry.fl)
            aircraft[callsign] = Aircraft(flight_plan, pos.lat, pos.lon, pos.fl, heading, speed, vertical_speed)

        return aircraft
